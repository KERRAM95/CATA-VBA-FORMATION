Attribute VB_Name = "Kin_Dressup_sample" 
'-------------- Sample MACRO written in Basic Script ------------------------------------------ 
' vba00000018_76A10EBA-0000-2314-5CF527E8000414A2 
'This sample retrieves a Mechanism from a Product 
'It calls the dressup methods GetDressupProducts, AttachDressupProduct , DetachDressupProduc 
' and the MehanismProducts Properties 
' 
'Before starting this macro, opens SliderCrankMechanism product 
'Steps: 
' 1 Gets the current editor 
' 2 Gets the product Service from the editor 
' 3 Gets all the occurrences under the root occurrence 
' 4 Gets the context 
' 5 Gets the discipline of all representation of the tree 
' 6 Gets the mechanism representation if the V_discipline is a mechanism 
' 7 Gets Part3.1, an occurrence of the mechanism 
' 8 Gets all the occurrences attached to Part3.1 
' 9 Detaches all the products previously attached to Part3.1 
' 10 Gets Product1.1 VPMOccurrence 
' 11 Attaches Product1.1 to Part3.1 
' 12 Lists all the products attached to Part3.1 
' 13 Lists all the mechanism products 
'----------------------------------------------------------------------------------------------- 

Option Explicit 

Sub CATMain() 

   ' 1 Gets the current editor 
   Dim oEditor As Editor 
   Set oEditor = CATIA.ActiveEditor 

   ' 2 Gets the product Service from the Editor 
   Dim MyPrdService As PLMProductService 
   Set MyPrdService = oEditor.GetService("PLMProductService") 
   Dim myRootOccurrence As VPMRootOccurrence 
   Set myRootOccurrence = MyPrdService.RootOccurrence 

   ' 3 Gets all the occurrences under the root occurrence 
   Dim cOccs As VPMOccurrences 
   Set cOccs = myRootOccurrence.Occurrences 

   ' 4 Gets the context 
   Dim MyContext 
   Set MyContext = oEditor.GetService("PLMProductContext") 

   ' 5 Gets the discipline of all representation of the tree 
   Dim MyEntity 
   For Each MyEntity In MyContext.EditedContent 
      Dim MyRef As VPMReference 
      Set MyRef = MyEntity 
      Dim MyReps As VPMRepInstances 
      Set MyReps = MyRef.RepInstances 
      Dim MyRep 
      For Each MyRep In MyReps 
         Dim MyRepRef As VPMRepReference 
         Set MyRepRef = MyRep.ReferenceInstanceOf 
         Dim attr As String 
         attr = MyRepRef.GetAttributeValue("V_discipline") 

         ' 6 Gets the mechanism representation if the V_discipline is a mechanism 
         If (attr = "Mechanism") Then 
            Dim MyMechRep As KinMechanism 
            Set MyMechRep = MyRepRef.GetItem("MECHRep") 

            MsgBox " The mechanism name is " & MyMechRep.Name 

            ' 7 Gets Part3.1, an occurrence of the mechanism 
            Dim Part31 As VPMOccurrence 
            Set Part31 = cOccs.GetItem("Part3.1") 

            ' 8 Gets all the occurrences attached to Part3.1 
            Dim ListAttached 
            ListAttached = MyMechRep.GetDressupProducts(Part31) 
            MsgBox " call of GetDressupProducts " 

            ' 9 Detaches all the products previously attached to Part3.1 
            Dim Maxi 
            Maxi = UBound(ListAttached) 
            MsgBox " Number of dressup attached product to Part3.1 = " & (Maxi + 1) 
            Dim j As Integer 
            For j = 0 To Maxi 
               Dim Product As VPMOccurrence 
               Set Product = ListAttached(j) 
               Call MyMechRep.DetachDressupProduct(Product) 
               MsgBox Product.Name & " is detached from " & Part31.Name 
            Next 

            ' 10 Gets Product1.1 VPMOccurrence 
            Dim Product11 As VPMOccurrence 
            Set Product11 = cOccs.GetItem("Product1.1") 
            MsgBox " Get Product1.1 " 

            ' 11 Attaches Product1.1 to Part3.1 
            Call MyMechRep.AttachDressupProduct(Part31, Product11) 
            MsgBox " Attach Product1.1 to Part3.1" 

            ' 12 Lists all the products attached to Part3.1 
            ListAttached = MyMechRep.GetDressupProducts(Part31) 
            Maxi = UBound(ListAttached) 
            MsgBox " Number of attached products to in Part3.1 the dressup = " & (Maxi + 1) 

            For j = 0 To Maxi 
               Set Product = ListAttached(j) 
               MsgBox " Attached product = " & Product.Name 
            Next 

            ' 13 Lists all the mechanism products 
            Dim MechProducts 
            MechProducts = MyMechRep.MechanismProducts 

            Dim Msg As String 
            Dim i As Integer 
            For i = 0 To UBound(MechProducts) 
               Dim MecProduct As VPMOccurrence 
               Set MecProduct = MechProducts(i) 
               Msg = Msg & MecProduct.Name 
            Next 
            MsgBox " The mechanism contains the following product : " & Msg 

         End If 
      Next 
   Next 

End Sub
