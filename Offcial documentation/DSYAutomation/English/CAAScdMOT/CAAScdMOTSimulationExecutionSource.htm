'-------------- Sample MACRO written in Visual Basic for Applications and Python ------------------------------------------ 
'This sample retrieves the simulation object from the editor 
'and run one or all the analysis cases.
'
'Before starting this macro, open a Motion simulation object containing one or several analysis case(s) under the scenario.
'Steps:
'   1 - Retrieve the current Editor
'   2 - Retrieve the Product Service from the Editor
'   3 - Retrieve the Entities from the Product Service
'   4 - Retrieve the simulation object
'   5 - Retrieve the Motion Execution Services from the Editor
'   6 - Run one analysis case
'   7 - Retrieve execution information
'   8 - Run all the analysis cases
'   9 - Retrieve execution information
VBA Python   
Sub CATMain()

' 1 - Retrieve the current Editor 
Dim myEditor As Editor
Set myEditor = CATIA.ActiveEditor

' 2 - Retrieve the Product Service from the Editor 
Dim myPrdService As PLMProductService
Set myPrdService = myEditor.GetService("PLMProductService")

' 3 - Retrieve the Entities from the Product Service
Dim myEntities As PLMEntities
Set myEntities = myPrdService.EditedContent

' 4 - Retrieve the simulation object
Dim mySimulationObject As SimulationReference
Set mySimulationObject = myEntities.Item(1)

' 5 - Retrieve the Motion Execution Services from the Editor
Dim myExecutionServices As MOTExecutionServices
Set myExecutionServices = myEditor.GetService("MOTExecutionServices")

' 6 - Run one analysis case
Dim inputAnalysisCase As String
inputAnalysisCase = "Motion Analysis Case.1"
Dim analysisCaseExecutionInfos As MOTAnalysisCaseExecutionInfos
Call myExecutionServices.RunAnalysisCase(mySimulationObject, inputAnalysisCase, analysisCaseExecutionInfos)

' 7 - Retrieve execution information
Dim analysisCaseName As String
analysisCaseName = analysisCaseExecutionInfos.AnalysisCaseName

Dim execStatus As MOTExecutionStatus
execStatus = analysisCaseExecutionInfos.ExecutionStatus

Dim executionLog As String
executionLog = analysisCaseExecutionInfos.ExecutionLog

' 8 - Run all the analysis cases
Dim listOfAnalysisCaseExecutionInfos()
Call myExecutionServices.RunAllAnalysisCases(mySimulationObject, listOfAnalysisCaseExecutionInfos)

' 9 - Retrieve execution information
For Each execInfos In listOfAnalysisCaseExecutionInfos
  Dim analysisCase As String
  analysisCase = execInfos.AnalysisCaseName

  Dim status As MOTExecutionStatus
  status = execInfos.ExecutionStatus

  Dim log As String
  log = execInfos.ExecutionLog
Next execInfos

End Sub

import sys, os

myEditor = CATIA.ActiveEditor
