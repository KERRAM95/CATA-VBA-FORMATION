' <!-- -------------- Sample macro written in VB ------------------------------------------ -->
' <!-- This sample is an automated modeling procedure to creates rule based 3D octree Mesh part. --> 

' <!-- Steps :--> 
' <!-- 1 - Inputs definitions --> 
' <!-- 2 - VB Script to create Mesh Part --> 
    ' <!-- a - Retrieve created FEM Rep--> 
    ' <!-- b - Create Mesh Part --> 
    ' <!-- c - Create Solid Property --> 
    ' <!-- d - Retrieve Mesh Part Support if it is not given as argument --> 
    ' <!-- e - Add support to Mesh part --> 
    ' <!-- f - Set Rule to Mesh part --> 

' <!-- -------------------------------------------------------------------------- --> 
' <!-- 1 - Inputs definitions --> 
' <!-- -------------------------------------------------------------------------- --> 


'<Inputs>
  '<Input Name = "Support" Type ="Port" Mand ="false" Nls ="Support"/>
  '<Input Name = "MeshingRule" Type ="DmtDocument" SubType ="CATFmtOct3DRule" Mand ="true" Nls ="Rule"/>
  '<Input Name = "Property" Type ="Boolean" Mand ="true" Nls ="CreateSolid"/>
'</Inputs>

' <!-- ---------------------------------------------------------------------------------------------------- -->
' <!-- 2 - VB Script to create Mesh Part --> 
' <!-- ---------------------------------------------------------------------------------------------------- --> 

Sub main(iSupport, iRule, iProp)

' a - Retrieve created FEM Rep
'----------------------------------------------------------------------------------------------------
  Dim FEMRoot As CATIAFmtFemRoot
  Set FEMRoot = FEM.GetItem("SimFemRoot")

' b - Create Mesh Part
'----------------------------------------------------------------------------------------------------
  Dim MeshSet As CATIAFmtMeshSet
  Set MeshSet = FEMRoot.GetSet("SimNodesElements")
  Dim MeshParts As CATIAFmtMeshParts
  Set MeshParts = MeshSet.MeshParts
  Dim MeshPart As CATIAFmtMeshPart
  Set MeshPart = MeshParts.Add("CATFmtOctree3DRulesMesher")

' c - Create Solid Property
'----------------------------------------------------------------------------------------------------
'
' if requested Create Solid Property
'
  if (iProp = "True") Then 
    Dim PropertySet As SimProperties
    Set PropertySet = FEMRoot.GetSet("SimProperties")
    Dim SolidSection As SimSolidProperty
    Set SolidSection = PropertySet.Add("SMAMpaSolidProperty")
  end If

  Dim LinkAccess As SimLinkAccess
 
' d - Retrieve Mesh Part Support if it is not given as argument
'----------------------------------------------------------------------------------------------------
  if IsEmpty(iSupport) Then
    Dim ShapeRef As VPMRepReference
    Set ShapeRef = SHAPE.ReferenceInstanceOf
    Dim ThePart As Part
    Set ThePart = ShapeRef.GetItem("Part")
    Dim PartBody As AnyObject
    Set PartBody = ThePart.MainBody
    Dim Testvar As Boolean
    Testvar = IsAnEmptyBody(PartBody)
    If Testvar = True Then
      Dim myOrderedGeometricalSets As OrderedGeometricalSets
      Set myOrderedGeometricalSets = ThePart.OrderedGeometricalSets
      If myOrderedGeometricalSets.Count = 0 Then
          Set PartBody = ThePart.MainBody
      Else: Set PartBody = myOrderedGeometricalSets.Item(1)
      End If  
    End If
' e - Add support to Mesh part
'----------------------------------------------------------------------------------------------------
    Dim Ref As Reference
    Set Ref = ThePart.CreateReferenceFromObject(PartBody)
    Dim EmptyOcc As PLMOccurrence
    Set EmptyOcc = OCCURRENCE
    Dim FEMService As SimComposeLinkServices
    Set FEMService = FEM.GetItem("SimComposeLinkServices")
    Set Link = FEMService.ComposeLink(EmptyOcc, SHAPE, Ref)
    MeshPart.AddSupport (Link)
    if (iProp = "True") Then 
      Set LinkAccess = SolidSection.GetItem("SimLinkAccess")
      LinkAccess.AddLink "MainSupport", Link
    end if
  else
    MeshPart.AddSupport (iSupport)
    if (iProp = "True") Then 
      Set LinkAccess = SolidSection.GetItem("SimLinkAccess")
      LinkAccess.AddLink "MainSupport", iSupport
    end if
  end If

' f - Set Rule to Mesh part
'----------------------------------------------------------------------------------------------------
  MeshPart.SetRule iRule,"CATFmtOct3DRule"

End Sub

Private Function IsAnEmptyBody(ByVal iBody As Body) As Boolean

If iBody Is Nothing Then
  Exit Function
Else
  Dim myShapes As Shapes
  Set myShapes = iBody.Shapes
  Dim myHybridShapes As HybridShapes
  Set myHybridShapes = iBody.HybridShapes
  IIsAnEmptyBody = True
  If Not (myShapes Is Nothing) Then
    If myShapes.Count > 0 Then
      IsAnEmptyBody = False
    End If
  End If
  If Not (myHybridShapes Is Nothing) Then
    If myHybridShapes.Count > 0 Then
      IsAnEmptyBody = False
    End If
  End If
End If

End Function
