' <!-- -------------- Sample macro written in VB ------------------------------------------ -->
' <!-- This sample is an automated modeling procedure to creates parameter based surface quad Mesh part. --> 

' <!-- Steps :--> 
' <!-- 1 - Inputs definitions --> 
' <!-- 2 - VB Script to create Mesh Part --> 
	' <!-- a - Retrieve created FEM Rep--> 
	' <!-- b - Create Mesh Part --> 
	' <!-- c - Create Shell Property --> 

' <!-- -------------------------------------------------------------------------- --> 
' <!-- 1 - Inputs definitions --> 
' <!-- -------------------------------------------------------------------------- --> 
'<Lists>
  '<List Name = "ThicknessDefinition">
    '<Member Name = "NoShellSection" Nls = "NoShellSection"/>
    '<Member Name = "Specified" Nls = "Specified"/>
    '<Member Name = "FromSolid" Nls = "FromSolid"/>
    '<Member Name = "FromThinPartFeature" Nls = "FromThinPartFeature"/>
   '</List>
   '<List Name = "SelectionDefinition">
    '<Member Name = "MainBody" Nls = "MainBody"/>
    '<Member Name = "AllGeometries" Nls = "AllGeometries"/>
    '<Member Name = "VisibleGeometries" Nls = "VisibleGeometries"/>
   '</List>
'</Lists>
'<Enums>
  '<Enum Name = "ElementOrder">
    '<Member Name = "Linear"    Nls ="Linear"/>
    '<Member Name = "Quadratic" Nls ="Quadratic"/>
   '</Enum>
'</Enums>
'<Inputs>
  '<Input Name = "SelectionMode" Type = "List" SubType = "SelectionDefinition" Mand ="false" Nls ="SelectionMode">
	'<Input Name = "Support" Type ="Port" Visible = "1" Mand ="true" Nls ="Support"/>
  '</Input>
  '<Input Name = "MeshSize" Type ="Dimension" SubType ="Length" Mand ="true" Nls ="MeshSize"/>
  '<Input Name = "ElementOrder" Type ="Enum" SubType ="ElementOrder" Mand ="true" Nls ="ElementOrder"/>
  '<Input Name = "ShellSection" Type = "List" SubType = "ThicknessDefinition" Mand ="true" Nls ="CreateShellSection">
    '<Input Name = "Specified" Type = "Dimension" SubType = "Length" Visible = "2" Mand ="true" Nls ="Specified" />
    '<Input Name = "FromSolid" Type = "Selection" SubType = "MonoSelectionSolids" Visible = "3" Mand ="true" Nls ="FromSolid" />
  '</Input>
'</Inputs>

' <!-- ---------------------------------------------------------------------------------------------------- -->
' <!-- Remark on User-Driven Inputs definitions --> 
' <!-- ---------------------------------------------------------------------------------------------------- --> 
' <!-- In User-Driven method there is an additional input FromSolid for Shell sections definition --> 

' <!-- ---------------------------------------------------------------------------------------------------- -->
' <!-- 2 - VB Script to create Mesh Part --> 
' <!-- ---------------------------------------------------------------------------------------------------- --> 

Sub main(iMode, iSupport, iSize, iOrder, iProp, iSpecified, iSelection)
  
 'Get Shape Ref and Part
 Dim MyRepRef As VPMRepReference
 Set MyRepRef = Shape.ReferenceInstanceOf
 Dim MyPart As Part
 Set MyPart = MyRepRef.GetItem("Part")
 'Get the FEM
 Dim MyFEMRoot As SimFemRoot
 Set MyFEMRoot = FEM.GetItem("SimFemRoot")
 ' Get AutomatedFEM Services
 Dim AutomatedFEMServices As SimAutoModServices
 Set AutomatedFEMServices = CATIA.ActiveEditor.GetService("SimAutoModServices")
 'Get the PLM Product Service
 Dim MyProductService As PLMProductService
 Set MyProductService = CATIA.ActiveEditor.GetService("PLMProductService")	
  
 'Mode Single Body 
 If (iMode = "1") Then
   CreateMeshPart MyFEMRoot, iSupport, iSize, iOrder, iProp, iSpecified, iSelection
 End If 
  
 Dim CheckVisibility As Boolean
 CheckVisibility = False
 If(iMode = "3") Then
   CheckVisibility = True
 End If 
 Dim myListSupport
 Dim countSupport As Integer
 countSupport = 0
 'Mode Multiple/Visible Bodies
 If (iMode = "2") or (iMode = "3") Then
   'Get all 2D results in 3DPart
   Dim myListResults
   On Error Resume Next
   myListResults = AutomatedFEMServices.GetAllResults(MyPart,2)
   If Err = 0 Then
     BuildListSupport myListResults, CheckVisibility, myListSupport
     countSupport = UBound(myListSupport) - LBound(myListSupport) + 1
   End If 
    On Error GoTo 0
    If countSupport = 0 Then
      Exit Sub 
    End If
	
   'Loop on supports
   Dim RefSupport As Reference
   For i = 0 To countSupport - 1
     Dim Support As Body
     Set Support = myListSupport(i)
     Set RefSupport = MyPart.CreateReferenceFromObject(Support)
     Set Link = MyProductService.ComposeLink(OCCURRENCE, SHAPE, RefSupport)
     CreateMeshPart MyFEMRoot, Link, iSize, iOrder, iProp, iSpecified, iSelection
   Next
	
 End If 
  
End Sub

Function StringToDouble (str As String) As Double
  For i=1 To Len(str)
	If Not IsNumeric(Mid(str,i,1)) And Not Mid(str,i,1)="." And Not Mid(str,i,1)="," Then
      StringToDouble=CDbl(Mid(str,1,Len(str)-2))*0.001
    End If
  Next
End Function

Private Sub BuildListSupport(ByVal myListResults As AnyObject, ByVal CheckVisibility As Boolean, ByRef myListSupport As AnyObject)

 Dim countResults As Integer
 countResults = UBound(myListResults) - LBound(myListResults) + 1
 Dim countSupport As Integer
 countSupport = 0
 ReDim myListSupport(countResults)
	
 'Loop on 3D Results
 For i = 0 To countResults - 1
   Dim Support As Body 
   Set Support = myListResults(i)
   Dim Mysel As Selection
   Set Mysel = CATIA.ActiveEditor.Selection
   Dim visStatus As CatVisPropertyShow
   Mysel.Add Support
   Mysel.VisProperties.GetShow visStatus
   Mysel.Clear
   If (CheckVisibility = False) or ((CheckVisibility = True) and (visStatus = catVisPropertyShowAttr)) Then
     Set myListSupport(countSupport) = Support
     countSupport = countSupport + 1
   End If 
 Next

 'Resize arrays
 countSupport = countSupport - 1
 ReDim Preserve myListSupport(countSupport)
	
End Sub

Private Sub CreateMeshPart(ByVal FEMRoot As CATIAFmtFemRoot, ByVal Support As AnyObject, ByVal iSize As String, ByVal iOrder As Integer, ByVal iProp As Integer, ByVal iSpecified As  String, ByVal iSelection As AnyObject)

 Dim MeshSet As CATIAFmtMeshSet
 Set MeshSet = FEMRoot.GetSet("SimNodesElements")
 Dim MyMeshParts As CATIAFmtMeshParts
 Set MyMeshParts = MeshSet.MeshParts
 Dim MyMeshPart As CATIAFmtMeshPart
 Set MyMeshPart = MyMeshParts.Add("CATFmtSurfQuadMesher")
 MyMeshPart.AddSupport (Support)
 MyMeshPart.SetAttributeValue "Mesh", "MeshSizeValue", iSize
 MyMeshPart.SetAttributeValue "Mesh", "ElementOrder", iOrder
 '
 'If requested Create Shell Property
 '
 If (iProp > "1") Then
   Dim PropertySet As SimProperties
   Set PropertySet = FEMRoot.GetSet("SimProperties")
   Dim ShellSection As SimShellSection
   Set ShellSection = PropertySet.Add("SimShellSection")
   Dim LinkAccess As SimLinkAccess
   Set LinkAccess = ShellSection.GetItem("SimLinkAccess")
   LinkAccess.AddLink "MainSupport", Support

   If (iProp = "2") Then
     ShellSection.ThicknessType = SimShellSectionUser
     ShellSection.UniformThickness = StringToDouble(iSpecified)
   End If
   If (iProp = "3") Then
     ShellSection.ThicknessType = SimShellSectionMid
     For Each selection In iSelection
       LinkAccess.AddLink "SecondarySupport", selection
     Next
   End If
   If (iProp = "4") Then    
     ShellSection.ThicknessType = SimShellSectionThin
   End If
 End If
	
End Sub
