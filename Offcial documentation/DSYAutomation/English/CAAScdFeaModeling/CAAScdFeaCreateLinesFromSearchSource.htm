'-------------- Sample MACRO written in Basic Script ------------------------------------------
'This sample creates line fasteners by giving an handlers list containing lines.

'Steps:
'    1- Retrieves the active editor and the root product
'    2- Retrieves the root occurrence
'    3- Retrieves the active product
'    4- Retrieves the rep instance
'    5- Retrieves the FEM Representation and the part
'    6- Retrieves the connection services object
'    7- Get the geometrical set named : Weld Edges for CAE 
'    8- Creates Line fasteners from search 

Sub CATMain()

'--------------------------------------------------------------------------
' 1- Retrieves the active editor and the root product
'--------------------------------------------------------------------------
Dim myEditorAs Editor
Set myEditor = CATIA.ActiveEditor

Dim myPLMServicesAs PLMProductService
Set myPLMServices = myEditor.GetService("PLMProductService")

Dim myEntitiesAs PLMEntities
 Set myEntities = myPLMServices.EditedContent

Dim myRootProductAs VPMReference
Set myRootProduct = myEntities.Item(1)
'--------------------------------------------------------------------------
' 2- Retrieves the root occurrence
'--------------------------------------------------------------------------
Dim MyRootOccAs VPMRootOccurrence
Set MyRootOcc = myEditor.ActiveObject

'--------------------------------------------------------------------------
' 3- Retrieves the active product
'--------------------------------------------------------------------------
Dim MyActiveProductAs VPMReference
Set MyActiveProduct = MyRootOcc.PLMEntity

'--------------------------------------------------------------------------
' 4- Retrieves the rep instances
'--------------------------------------------------------------------------
Dim MyVPMRepInstancesAs VPMRepInstances
Set MyVPMRepInstances = MyActiveProduct.RepInstances
'--------------------------------------------------------------------------
' 5- Retrieves the FEM Representation and the part
'--------------------------------------------------------------------------
 Dim attr As String
 Dim myRepRef As VPMRepReference
 Dim MyFemRepRoot As SimFemRoot

 For Each MyEntity In myPLMServices.EditedContent
     Dim MyRef As VPMReference
     Set MyRef = MyEntity

     For Each myRep In MyVPMRepInstances
         Set myRepRef = myRep.ReferenceInstanceOf
         attr = myRepRef.GetAttributeValue("V_discipline")
             If (attr = "FEM") Then
                 Set MyFemRepRoot = myRepRef.GetItem("SimFemRoot")
             End If
             If (attr = "Design") Then
                Set MyPart = myRepRef.GetItem("Part")
             End If
     Next
 Next
'--------------------------------------------------------------------------
' 6- Retrieves the connection services object
'--------------------------------------------------------------------------
  Dim myConnectionService As SimConnectionServices

  Set myConnectionService = MyFemRepRoot.GetItem("SimConnectionServices")
 
'--------------------------------------------------------------------------
' 7- Get the geometrical set named : Weld Edges for CAE
'--------------------------------------------------------------------------
  Dim hander(0)
  Set hander(0) = MyPart.FindObjectByName("Weld Edges for CAE")

'--------------------------------------------------------------------------
' 8- Creates Line fasteners from search
'--------------------------------------------------------------------------
  myConnectionService.CreateLinesFromSearch hander, "Test_", 10.0

End Sub
