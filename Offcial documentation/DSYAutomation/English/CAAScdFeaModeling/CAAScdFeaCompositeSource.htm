'-------------- Sample MACRO written in Basic Script ------------------------------------------
'This sample demonstrates the methodology to create/edit composite features.

'Steps:
	Setting up the Model
	Managing the Composite Parameters
	Adding materials to the Composite Parameters
	Managing Rosettes in the Composite Parameters
	Add Orientation in the Composite Parameters
	Managing Laminates in the Composite Parameters
	Creation of Grid using reference geometries and Rosette
	Assigning laminate to Grid Cells
	Renaming the Grid Cells
	Exporting the Grid to a file


Sub CATMain()

'--------------------------------------------------------------------------
' 1- Setting up the model
'--------------------------------------------------------------------------
    Dim matSearchService As SearchService
    Set matSearchService = CATIA.GetSessionService("Search")
    Dim oDatabaseSearch1 As DatabaseSearch
    Set oDatabaseSearch1 = matSearchService.DatabaseSearch
    oDatabaseSearch1.BaseType = "PLMEntity"
    oDatabaseSearch1.Mode = SearchMode_Predefined
    oDatabaseSearch1.AddPredefinedCriteria "mat:*Com*VB*"
    matSearchService.Search
    Dim matPLMEntities As PLMEntities
    Set matPLMEntities = oDatabaseSearch1.Results
    Dim matPLMOpenService As PLMOpenService
    Set matPLMOpenService = CATIA.GetSessionService("PLMOpenService")
    Dim oEditor As Editor
    For Each i In matPLMEntities
        matPLMOpenService.PLMOpen i, oEditor
    Next

    Dim oSearchService As SearchService
    Set oSearchService = CATIA.GetSessionService("Search")
    Dim oDatabaseSearch As DatabaseSearch
    Set oDatabaseSearch = oSearchService.DatabaseSearch
    oDatabaseSearch.BaseType = "VPMReference"
    oDatabaseSearch.AddEasyCriteria "V_Name", "VB*"
    oSearchService.Search
    Dim cPLMEntities As PLMEntities
    Set cPLMEntities = oDatabaseSearch.Results
    Dim oPLMOpenService As PLMOpenService
    Set oPLMOpenService = CATIA.GetSessionService("PLMOpenService")
    Dim Editor As Editor
    Set i = cPLMEntities.Item(1)
    oPLMOpenService.PLMOpen i, Editor
    
    Dim myActiveObject As Part
    Set myActiveObject = Editor.ActiveObject
    
    Dim oProductService   As PLMProductService
    Set oProductService = Editor.GetService("PLMProductService")
    
    Dim myEntities As PLMEntities
    Set myEntities = oProductService.EditedContent
    
    Dim myModel As VPMReference
    Set myModel = myEntities.Item(1)
    
    ' Retrieve the root occurrence
    Dim myRootOcc As VPMRootOccurrence
    Set myRootOcc = oProductService.RootOccurrence
    ' retrieve the publications set
    Dim myPublications As VPMPublications
    Set myPublications = myRootOcc.ReferenceRootOccurrenceOf.Publications
    'Main supports
    Dim Rib01 As VPMPublication
    Set Rib01 = myPublications.GetItem("Rib01")
    Dim Rib02 As VPMPublication
    Set Rib02 = myPublications.GetItem("Rib02")
    Dim Rib03 As VPMPublication
    Set Rib03 = myPublications.GetItem("Rib03")
    Dim Rib04 As VPMPublication
    Set Rib04 = myPublications.GetItem("Rib04")
    Dim Rib05 As VPMPublication
    Set Rib05 = myPublications.GetItem("Rib05")
    Dim Skin As VPMPublication
    Set Skin = myPublications.GetItem("Skin")
    

    Dim myRootOccurrence As VPMRootOccurrence
    Set myRootOccurrence = oProductService.RootOccurrence
    
    Dim myRepInstance As VPMRepInstance
    Set myRepInstance = Nothing ' Because we use publications
    
    Dim myLinkToSupport1 As AnyObject
    Set myLinkToSupport1 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Rib01)
    Dim myLinkToSupport2 As AnyObject
    Set myLinkToSupport2 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Rib02)
    Dim myLinkToSupport3 As AnyObject
    Set myLinkToSupport3 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Rib03)
    Dim myLinkToSupport4 As AnyObject
    Set myLinkToSupport4 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Rib04)
    Dim myLinkToSupport5 As AnyObject
    Set myLinkToSupport5 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Rib05)
    Dim myLinkToSupport6 As AnyObject
    Set myLinkToSupport6 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Skin)
    
    'Creating the FEM Rep and retrieving its Rep Manager
    
    Dim myPrdRepFactory As SimPrdRepFactory
    Set myPrdRepFactory = myModel.GetItem("SimPrdRepFactory")
    
    Dim myRepRef As VPMRepReference
    Set myRepRef = myPrdRepFactory.CreatePrdRep("FEM")
    
    Dim myFEMRoot As SimFemRoot
    Set myFEMRoot = myRepRef.GetItem("SimFemRoot")
    
    myFEMRoot.AddAssociatedRep myLinkToSupport1
    
    
    'Creating a Mesh on a Part
    
    Dim myMeshSet As SimMeshSet
    Set myMeshSet = myFEMRoot.GetSet("SimNodesElements")
    
    Dim myMeshParts As SimMeshParts
    Set myMeshParts = myMeshSet.MeshParts
    
    Dim myMeshPart1 As SimMeshPart
    Set myMeshPart1 = myMeshParts.Add("CATFmtSurfaceMesher")
    myMeshPart1.AddSupport myLinkToSupport1
    myMeshPart1.SetAttributeValue "Mesh", "ElementOrder", 1
    myMeshPart1.SetAttributeValue "Mesh", "MeshSizeValue", 40#
    
    Dim myMeshPart2 As SimMeshPart
    Set myMeshPart2 = myMeshParts.Add("CATFmtSurfaceMesher")
    myMeshPart2.AddSupport myLinkToSupport2
    myMeshPart2.SetAttributeValue "Mesh", "ElementOrder", 1
    myMeshPart2.SetAttributeValue "Mesh", "MeshSizeValue", 40#
     
    Dim myMeshPart3 As SimMeshPart
    Set myMeshPart3 = myMeshParts.Add("CATFmtSurfaceMesher")
    myMeshPart3.AddSupport myLinkToSupport3
    myMeshPart3.SetAttributeValue "Mesh", "ElementOrder", 1
    myMeshPart3.SetAttributeValue "Mesh", "MeshSizeValue", 40#
     
    Dim myMeshPart4 As SimMeshPart
    Set myMeshPart4 = myMeshParts.Add("CATFmtSurfaceMesher")
    myMeshPart4.AddSupport myLinkToSupport4
    myMeshPart4.SetAttributeValue "Mesh", "ElementOrder", 1
    myMeshPart4.SetAttributeValue "Mesh", "MeshSizeValue", 40#
     
    Dim myMeshPart5 As SimMeshPart
    Set myMeshPart5 = myMeshParts.Add("CATFmtSurfaceMesher")
    myMeshPart5.AddSupport myLinkToSupport5
    myMeshPart5.SetAttributeValue "Mesh", "ElementOrder", 1
    myMeshPart5.SetAttributeValue "Mesh", "MeshSizeValue", 40#
    myMeshSet.Update
     
    Dim myMeshPart6 As SimMeshPart
    Set myMeshPart6 = myMeshParts.Add("CATFmtSurfaceMesher")
    myMeshPart6.AddSupport myLinkToSupport6
    myMeshPart6.SetAttributeValue "Mesh", "ElementOrder", 1
    myMeshPart6.SetAttributeValue "Mesh", "MeshSizeValue", 40#
    myMeshSet.Update
	
    'Creating a Property
    
    Dim myPropertiesSet As SimProperties
    Set myPropertiesSet = myFEMRoot.GetSet("SimProperties")

    Dim myCompositeShellSection1 'As SimCompositeShellSection
    Set myCompositeShellSection1 = myPropertiesSet.Add("SimCompositeShellSection")
    Dim myLinkAccess1 As SimLinkAccess
    Set myLinkAccess1 = myCompositeShellSection1.GetItem("SimLinkAccess")
    myLinkAccess1.AddLink "MainSupport", myLinkToSupport1
'--------------------------------------------------------------------------
' 2- Managing the Composite Parameters
'--------------------------------------------------------------------------
    
    Dim myCompParams As SimCompositeParameters
    Set myCompParams = myCompositeShellSection1.GetCompositeParameters
'--------------------------------------------------------------------------
' 3- Adding materials to the Composite Parameters
'-------------------------------------------------------------------------- 
    Dim MatService As MATPLMService
    Set MatService = CATIA.GetSessionService("MATPLMService")
    Dim ListMaterialInSession As ListObject
    MatService.GetMaterialsInSession ListMaterialInSession
    Dim mat As Material
    Dim j
    For j = 1 To ListMaterialInSession.Count
        Set mat = ListMaterialInSession.Item(j)
        myCompParams.AddMaterial mat
    Next
'--------------------------------------------------------------------------
' 4- Managing Rosette in the Composite Parameters 
'--------------------------------------------------------------------------
    
    Dim myRosette1 As SimCompositeRosette
    Set myRosette1 = myCompParams.CreateRosette
    Set myAxisSystem1 = myActiveObject.FindObjectByName("Axis System.1")
    Set myAxisSystem3 = myActiveObject.FindObjectByName("Axis System.5")
    myRosette1.SetMainAxis myAxisSystem1
    myRosette1.SetRosetteTransferType SimCompositeRosetteTypeCartesian
    Dim myRosette2 As SimCompositeRosette
    Set myRosette2 = myCompParams.CreateRosette
    Set myAxisSystem2 = myActiveObject.FindObjectByName("Axis System.2")
    myRosette2.SetMainAxis myAxisSystem2
	
    'Geometry for Guided by curve Rosette Transfer Type
	
    myRosette2.SetRosetteTransferType SimCompositeRosetteTypeGuidedCurve
    Dim MyRosette2Cyl 'As SimCompositeRosetteType
    Set MyRosette2Cyl = myRosette2.GetRosetteCylindericalTransfer
    MyRosette2Cyl.SetCylindricalType SimCompositeCylindericalZero
    Set GeomSetRib01 = myActiveObject.FindObjectByName("Rib01")
    Dim GeomSetRib01Elems '  As HybridShapes
    Set GeomSetRib01Elems = GeomSetRib01.HybridShapes
    Dim Val ' As HybridShapeExtract
    Set Val = GeomSetRib01Elems.Item(1)
    MyRosette2Cyl.SetCylindricalType SimCompositeCylindericalZero
    MyRosette2Cyl.SetCylindricalRosetteTranferCurve Val
    Dim myRosette3 As SimCompositeRosette
    Set myRosette3 = myCompParams.CreateRosette
    myRosette3.SetMainAxis myAxisSystem3
    myRosette3.SetRosetteTransferType SimCompositeRosetteTypeCartesian

'--------------------------------------------------------------------------
' 5- Add Orientation in the Composite Parameters
'--------------------------------------------------------------------------
    myCompParams.CreateOrientation "Simulia", 37.5, 1, 2, 3
    
'--------------------------------------------------------------------------
' 6- Managing Laminates in the Composite Parameters
'--------------------------------------------------------------------------
    Dim mats
    mats = myCompParams.GetMaterialList
    Dim listSize As Long
    listSize = UBound(mats) - LBound(mats) + 1
    ReDim MyParam(LBound(mats) To UBound(mats))
    For h = LBound(mats) To UBound(mats)
        MyParam(h) = 45
    Next
    Dim lamina 'As SimCompositeLaminate
    Set lamina = myCompParams.CreateLaminate("Lamina1")
    lamina.SetStackingSequence mats, MyParam
	
'--------------------------------------------------------------------------
' 7- Creation of Grid using reference geometries and Rosette
'--------------------------------------------------------------------------   
    Set GeomSetRib01 = myActiveObject.FindObjectByName("Rib01")
    Dim size As Long
    size = GeomSetRib01Elems.Count
    ReDim myGeomRefSetForGridRib01(size - 1) 'As ListObject
    'Dim Val As HybridShapeExtract
    For hh = 0 To size - 1
        Set Val = GeomSetRib01Elems.Item(hh + 1)
        Set myGeomRefSetForGridRib01(hh) = Val
    Next

    Set Grid1 = myCompositeShellSection1.CreateGrid(myGeomRefSetForGridRib01, myRosette1)
	
'--------------------------------------------------------------------------
' 8- Assigning Laminate to Grid Cells
'--------------------------------------------------------------------------   	
    Dim GriedCells1
    GriedCells1 = Grid1.GetCells
    For Each C In GriedCells1
        C.SetLaminateOnCell lamina
    Next
'--------------------------------------------------------------------------
' 9- Renaming the Grid Cells
'-------------------------------------------------------------------------- 
    Dim names
    For Each C In GriedCells1
        names = C.GetCellName
        names = "1" + names
        C.SetCellName names
    Next
	
	'Similarly Composite Grid feature can be created for rest of the geometries
    Dim myLinkToCPDSupport1 As AnyObject
    Set myLinkToCPDSupport1 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Grid1)
    Dim myCPDSupport1 As SimCompositeSupportDefinition
    Set myCPDSupport1 = myCompositeShellSection1.CompositeSupport
    Dim mycpdLinkAccess1 As SimLinkAccess
    Set mycpdLinkAccess1 = myCPDSupport1.GetItem("SimLinkAccess")
    mycpdLinkAccess1.AddLink "ConnectorList", myLinkToCPDSupport1
        
    Dim myCompositeShellSection2 'As SimCompositeShellSection
    Set myCompositeShellSection2 = myPropertiesSet.Add("SimCompositeShellSection")
    Dim myLinkAccess2 As SimLinkAccess
    Set myLinkAccess2 = myCompositeShellSection2.GetItem("SimLinkAccess")
    myLinkAccess2.AddLink "MainSupport", myLinkToSupport2
    
    Set GeomSetRib02 = myActiveObject.FindObjectByName("Rib02")
    Dim GeomSetRib02Elems  As HybridShapes
    Set GeomSetRib02Elems = GeomSetRib02.HybridShapes
    size = GeomSetRib02Elems.Count
    ReDim myGeomRefSetForGridRib02(size - 1) 'As ListObject
    For hh = 0 To size - 1
        Set Val = GeomSetRib02Elems.Item(hh + 1)
        Set myGeomRefSetForGridRib02(hh) = Val
    Next
    
    Set Grid2 = myCompositeShellSection2.CreateGrid(myGeomRefSetForGridRib02, myRosette1)
    Dim GriedCells2
    GriedCells2 = Grid2.GetCells
    For Each C In GriedCells2
        C.SetLaminateOnCell lamina
        names = C.GetCellName
        names = "2" + names
        C.SetCellName names
    Next
    Dim myLinkToCPDSupport2 As AnyObject
    Set myLinkToCPDSupport2 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Grid2)
    Dim myCPDSupport2 As SimCompositeSupportDefinition
    Set myCPDSupport2 = myCompositeShellSection2.CompositeSupport
    Dim mycpdLinkAccess2 As SimLinkAccess
    Set mycpdLinkAccess2 = myCPDSupport2.GetItem("SimLinkAccess")
    mycpdLinkAccess2.AddLink "ConnectorList", myLinkToCPDSupport2
    
    Dim myCompositeShellSection3 'As SimCompositeShellSection
    Set myCompositeShellSection3 = myPropertiesSet.Add("SimCompositeShellSection")
    Dim myLinkAccess3 As SimLinkAccess
    Set myLinkAccess3 = myCompositeShellSection3.GetItem("SimLinkAccess")
    myLinkAccess3.AddLink "MainSupport", myLinkToSupport3
    
    Set GeomSetRib03 = myActiveObject.FindObjectByName("Rib03")
    Dim GeomSetRib03Elems  As HybridShapes
    Set GeomSetRib03Elems = GeomSetRib03.HybridShapes
    size = GeomSetRib03Elems.Count
    ReDim myGeomRefSetForGridRib03(size - 1) 'As ListObject
    For hh = 0 To size - 1
        Set Val = GeomSetRib03Elems.Item(hh + 1)
        Set myGeomRefSetForGridRib03(hh) = Val
    Next
    Set Grid3 = myCompositeShellSection3.CreateGrid(myGeomRefSetForGridRib03, myRosette1)
    Dim GriedCells3
    GriedCells3 = Grid3.GetCells
    For Each C In GriedCells3
        C.SetLaminateOnCell lamina
        names = C.GetCellName
        names = "3" + names
        C.SetCellName names
    Next
    Dim myLinkToCPDSupport3 As AnyObject
    Set myLinkToCPDSupport3 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Grid3)
    Dim myCPDSupport3 As SimCompositeSupportDefinition
    Set myCPDSupport3 = myCompositeShellSection3.CompositeSupport
    Dim mycpdLinkAccess3 As SimLinkAccess
    Set mycpdLinkAccess3 = myCPDSupport3.GetItem("SimLinkAccess")
    mycpdLinkAccess3.AddLink "ConnectorList", myLinkToCPDSupport3
        
    Dim myCompositeShellSection4 'As SimCompositeShellSection
    Set myCompositeShellSection4 = myPropertiesSet.Add("SimCompositeShellSection")
    Dim myLinkAccess4 As SimLinkAccess
    Set myLinkAccess4 = myCompositeShellSection4.GetItem("SimLinkAccess")
    myLinkAccess4.AddLink "MainSupport", myLinkToSupport4
    
    Set GeomSetRib04 = myActiveObject.FindObjectByName("Rib04")
    Dim GeomSetRib04Elems  As HybridShapes
    Set GeomSetRib04Elems = GeomSetRib04.HybridShapes
    size = GeomSetRib04Elems.Count
    ReDim myGeomRefSetForGridRib04(size - 1) 'As ListObject
    For hh = 0 To size - 1
        Set Val = GeomSetRib04Elems.Item(hh + 1)
        Set myGeomRefSetForGridRib04(hh) = Val
    Next
    Set Grid4 = myCompositeShellSection4.CreateGrid(myGeomRefSetForGridRib04, myRosette1)
    Dim GriedCells4
    GriedCells4 = Grid4.GetCells
    For Each C In GriedCells4
        C.SetLaminateOnCell lamina
        names = C.GetCellName
        names = "4" + names
        C.SetCellName names
    Next
    Dim myLinkToCPDSupport4 As AnyObject
    Set myLinkToCPDSupport4 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Grid4)
    Dim myCPDSupport4 As SimCompositeSupportDefinition
    Set myCPDSupport4 = myCompositeShellSection4.CompositeSupport
    Dim mycpdLinkAccess4 As SimLinkAccess
    Set mycpdLinkAccess4 = myCPDSupport4.GetItem("SimLinkAccess")
    mycpdLinkAccess4.AddLink "ConnectorList", myLinkToCPDSupport4
        
    Dim myCompositeShellSection5 'As SimCompositeShellSection
    Set myCompositeShellSection5 = myPropertiesSet.Add("SimCompositeShellSection")
    Dim myLinkAccess5 As SimLinkAccess
    Set myLinkAccess5 = myCompositeShellSection5.GetItem("SimLinkAccess")
    myLinkAccess5.AddLink "MainSupport", myLinkToSupport5
    
    Set GeomSetRib05 = myActiveObject.FindObjectByName("Rib05")
    Dim GeomSetRib05Elems  As HybridShapes
    Set GeomSetRib05Elems = GeomSetRib05.HybridShapes
    size = GeomSetRib05Elems.Count
    ReDim myGeomRefSetForGridRib05(size - 1) 'As ListObject
    For hh = 0 To size - 1
        Set Val = GeomSetRib05Elems.Item(hh + 1)
        Set myGeomRefSetForGridRib05(hh) = Val
    Next
    Set Grid5 = myCompositeShellSection5.CreateGrid(myGeomRefSetForGridRib05, myRosette1)
    Dim GriedCells5
    GriedCells5 = Grid5.GetCells
    For Each C In GriedCells5
        C.SetLaminateOnCell lamina
        names = C.GetCellName
        names = "5" + names
        C.SetCellName names
    Next
    Dim myLinkToCPDSupport5 As AnyObject
    Set myLinkToCPDSupport5 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Grid5)
    Dim myCPDSupport5 As SimCompositeSupportDefinition
    Set myCPDSupport5 = myCompositeShellSection5.CompositeSupport
    Dim mycpdLinkAccess5 As SimLinkAccess
    Set mycpdLinkAccess5 = myCPDSupport5.GetItem("SimLinkAccess")
    mycpdLinkAccess5.AddLink "ConnectorList", myLinkToCPDSupport5
    
    Dim myCompositeShellSection6 'As SimCompositeShellSection
    Set myCompositeShellSection6 = myPropertiesSet.Add("SimCompositeShellSection")
    Dim myLinkAccess6 As SimLinkAccess
    Set myLinkAccess6 = myCompositeShellSection6.GetItem("SimLinkAccess")
    myLinkAccess6.AddLink "MainSupport", myLinkToSupport6
    
    Set GeomSetSkin = myActiveObject.FindObjectByName("Skin")
    Dim GeomSetSkinElems  As HybridShapes
    Set GeomSetSkinElems = GeomSetSkin.HybridShapes
    size = GeomSetSkinElems.Count
    ReDim myGeomRefSetForGridSkin(size - 1) 'As ListObject
    For hh = 0 To size - 1
        Set Val = GeomSetSkinElems.Item(hh + 1)
        Set myGeomRefSetForGridSkin(hh) = Val
    Next
    Set Grid6 = myCompositeShellSection6.CreateGrid(myGeomRefSetForGridSkin, myRosette1)
    Dim GriedCells6
    GriedCells6 = Grid6.GetCells
    For Each C In GriedCells6
        C.SetLaminateOnCell lamina
        names = C.GetCellName
        names = "6" + names
        C.SetCellName names
    Next
    Dim myLinkToCPDSupport6 As AnyObject
    Set myLinkToCPDSupport6 = oProductService.ComposeLink(myRootOccurrence, myRepInstance, Grid6)
    Dim myCPDSupport6 As SimCompositeSupportDefinition
    Set myCPDSupport6 = myCompositeShellSection6.CompositeSupport
    Dim mycpdLinkAccess6 As SimLinkAccess
    Set mycpdLinkAccess6 = myCPDSupport6.GetItem("SimLinkAccess")
    mycpdLinkAccess6.AddLink "ConnectorList", myLinkToCPDSupport6
    
    '--------------------------------------------------------------------------
    ' 10- Exporting the Grid to a file
    '-------------------------------------------------------------------------- 
    Grid1.CreateFileFromCells("C:\ExportedGrid.xls")

End Sub
