' <!-- -------------- Sample macro written in VB ------------------------------------------ --> 
' <!-- This sample is an automated modeling procedure to creates a parameter based 3D Tetrahedron Mesh part. --> 

' <!-- Steps : --> 

' <!-- 1 - Inputs definitions --> 
' <!-- 2 - VB Script to create Mesh Part --> 
    ' <!-- a - Retrieve Shape Reference and Part--> 
    ' <!-- b - Retrieve created FEM Rep and link services--> 
    ' <!-- c - Selection Mode = MainBody --> 
       ' <!-- Retrieve Mesh Part Support if it is not given as argument --> 
       ' <!-- Create Mesh Part and Solid Property --> 
    ' <!-- d - Selection Mode = All Geometries or All Visible Geometries -->
       ' <!-- Retrieve all 3D Results in Part --> 
       ' <!-- Loop on 3D Results  --> 

' <!-- -------------------------------------------------------------------------- --> 
' <!-- 1 - Inputs definitions --> 
' <!-- -------------------------------------------------------------------------- --> 


'<Lists>
  '<List Name = "SelectionDefinition">
    '<Member Name = "MainBody" Nls = "MainBody"/>
    '<Member Name = "AllGeometries" Nls = "AllGeometries"/>
    '<Member Name = "VisibleGeometries" Nls = "VisibleGeometries"/>
   '</List>
'</Lists>
'<Enums>
  '<Enum Name = "ElementOrder">
    '<Member Name = "Linear"    Nls ="Linear"/>
    '<Member Name = "Quadratic" Nls ="Quadratic"/>
   '</Enum>
'</Enums>
'<Inputs>
  '<Input Name = "SelectionMode" Type = "List" SubType = "SelectionDefinition" Mand ="false" Nls ="SelectionMode">
    '<Input Name = "Support" Type ="Port" Visible = "1" Mand ="false" Nls ="Support"/>
  '</Input>
  '<Input Name = "MeshSize" Type ="Dimension" SubType ="Length" Mand ="true" Nls ="MeshSize"/>
  '<Input Name = "ElementOrder" Type ="Enum" SubType ="ElementOrder" Mand ="true" Nls ="ElementOrder"/>
  '<Input Name = "SagValue" Type ="Dimension" SubType ="Length" Mand ="false" Nls ="SagValue"/>
  '<Input Name = "Property" Type ="Boolean" Mand ="true" Nls ="CreateSolid"/>
'</Inputs>


' <!-- ---------------------------------------------------------------------------------------------------- --> 
' <!-- 2 - VB Script to create Mesh Part --> 
' <!-- ---------------------------------------------------------------------------------------------------- --> 

Sub main(iMode, iSupport, iSize, iOrder, iSagValue, iProp)


' a - Retrieve Shape Reference and Part
'----------------------------------------------------------------------------------------------------
  Dim ShapeRef As VPMRepReference
  Set  ShapeRef = SHAPE.ReferenceInstanceOf
  Dim ThePart As Part
  Set  ThePart = ShapeRef.GetItem("Part")

' b - Retrieve created FEM Rep and link services
'----------------------------------------------------------------------------------------------------
  Dim FEMRoot As CATIAFmtFemRoot
  Set FEMRoot = FEM.GetItem("SimFemRoot")
  Dim EmptyOcc As PLMOccurrence
  Set EmptyOcc = OCCURRENCE
  Dim FEMService As SimComposeLinkServices
  Set FEMService = FEM.GetItem("SimComposeLinkServices")

' c - Selection Mode = MainBody
'----------------------------------------------------------------------------------------------------
  Dim EmptyBody As Boolean
  Dim RefSupport As Reference
  If (iMode = "1") Then
    If IsEmpty(iSupport) Then
      Dim PartBody As AnyObject
      Set PartBody = ThePart.MainBody
      EmptyBody = IsAnEmptyBody(PartBody,False)
      '-- Retrieve Mesh Part Support if it is not given as argument --
      If EmptyBody = True Then
        Set PartBody = FindFirstSolidBody(ThePart)
        If Not (PartBody Is Nothing) Then
          EmptyBody = False
        Else 
          Exit Sub 
        End If
      End If
      '-- Create link for Support and Mesh Part --
      Set RefSupport = ThePart.CreateReferenceFromObject(PartBody)
      Set Link = FEMService.ComposeLink(EmptyOcc, SHAPE, RefSupport)
      CreateMeshPart FEMRoot, EmptyOcc, ThePart, Link, iSize, iOrder, iProp, iSagValue
    Else
      '-- Create Mesh Part for iSupport --
      CreateMeshPart FEMRoot, EmptyOcc, ThePart, iSupport, iSize, iOrder, iProp, iSagValue
	End If
	 
  End If
    
' d - Selection Mode = All Geometries or All Visible Geometries -->
'----------------------------------------------------------------------------------------------------
  Dim count3DGeom As Integer
  count3DGeom = 0
  Dim CheckVisibility As Boolean
  CheckVisibility = False
  If (iMode = "3") Then
    CheckVisibility = True
  End If 

  If (iMode = "2") or (iMode = "3") Then
    Dim AutomatedFEMServices As SimAutoModServices
    Set AutomatedFEMServices = CATIA.ActiveEditor.GetService("SimAutoModServices")
    Dim myList3DGeom
    '-- Retrieve all 3D Results in Part --
    myList3DGeom = AutomatedFEMServices.GetAllResults(ThePart, 3)
    count3DGeom = UBound(myList3DGeom) - LBound(myList3DGeom) + 1
		
  '-- Loop on 3D Results --
  For i = 0 To count3DGeom - 1
    Dim Support As Body 
    Set Support = myList3DGeom(i)
    Dim Mysel As Selection
    Set Mysel = CATIA.ActiveEditor.Selection
    Dim visStatus As CatVisPropertyShow
    Mysel.Add Support
    Mysel.VisProperties.GetShow visStatus
    Mysel.Clear
    '-- Check visibility according to iMode --
    If (CheckVisibility = False) or ((CheckVisibility = True) and (visStatus = catVisPropertyShowAttr)) Then
      Set RefSupport = ThePart.CreateReferenceFromObject(Support)
      Set Link = FEMService.ComposeLink(EmptyOcc, SHAPE, RefSupport)
      '-- Create Mesh Part --
      CreateMeshPart FEMRoot, EmptyOcc, ThePart, Link, iSize, iOrder, iProp, iSagValue
    End If
   Next 
	
  End If

End Sub

Private Function IsAnEmptyBody(ByVal iBody As Body, ByVal iVisible As Boolean) As Boolean

If iBody Is Nothing Then
  Exit Function
Else
  Dim Mysel As Selection
  Set Mysel = CATIA.ActiveEditor.Selection
  Dim visStatus As CatVisPropertyShow
  Dim myShapes As Shapes
  Set myShapes = iBody.Shapes
  Dim myHybridShapes As HybridShapes
  Set myHybridShapes = iBody.HybridShapes
  IsAnEmptyBody = True
  If Not (myShapes Is Nothing) Then
    If myShapes.Count > 0 Then
      Mysel.Add iBody
      Mysel.VisProperties.GetShow visStatus
      Mysel.Clear
      If (iVisible = False) or ((iVisible = True) and (visStatus = catVisPropertyShowAttr)) Then
        IsAnEmptyBody = False
      End If
    End If
  End If
  If Not (myHybridShapes Is Nothing) Then
    If myHybridShapes.Count > 0 Then
      Mysel.Add iBody
      Mysel.VisProperties.GetShow visStatus
      Mysel.Clear
      If (iVisible = False) or ((iVisible = True) and (visStatus = catVisPropertyShowAttr)) Then
        IsAnEmptyBody = False
      End If
    End If
  End If
End If

End Function

' Find the fisrt volume in OGS
'----------------------------------------------------------------------------------------------------
Private Function FindFirstSolidBody(iPart) As Body

  Dim theMeasureService As MeasurableService
  Set theMeasureService = CATIA.ActiveEditor.GetService("MeasurableService")
  found = False
  For Each lOGS In iPart.OrderedGeometricalSets  
    For Each lHybShape In lOGS.HybridShapes
      Dim theMeasurableVolume As MeasurableVolume
      Set theMeasurableVolume = theMeasureService.GetMeasurable(lHybShape, CAAMeasurableVolume)
      Dim theArea As Double
      On Error Resume Next
      theArea = theMeasurableVolume.GetVolume
      If Err = 0 Then
        Set FindFirstSolidBody = lHybShape
        found = True
        Exit For
      End If
      On Error Goto 0
    Next
    If found=True Then Exit For
  Next
   On Error Goto 0
   If found=False Then Set FindFirstSolidBody=Nothing
   
End Function

' Create Mesh Part
'----------------------------------------------------------------------------------------------------
Private Sub CreateMeshPart(ByVal FEMRoot As CATIAFmtFemRoot, ByVal EmptyOcc As PLMOccurrence, ByVal ThePart As Part, ByVal Support As Body, ByVal iSize As String, ByVal iOrder As Integer, ByVal iProp As Integer, ByVal iSagValue As String)

  ' Create Mesh Part
  Dim MeshSet As CATIAFmtMeshSet
  Set MeshSet = FEMRoot.GetSet("SimNodesElements")
  Dim MeshParts As CATIAFmtMeshParts
  Set MeshParts = MeshSet.MeshParts
	
  Dim MeshPart As CATIAFmtMeshPart
  Set MeshPart = MeshParts.Add("CATFmtTetRulesMesher")
  ' Add Support to Mesh Part
  MeshPart.AddSupport (Support)
  ' If requested Create Solid Property
  If (iProp = 1) Then
    Dim PropertySet As SimProperties
    Set PropertySet = FEMRoot.GetSet("SimProperties")
    Dim SolidSection As SimSolidSection
    Dim LinkAccess As SimLinkAccess
    Set SolidSection = PropertySet.Add("SMAMpaSolidProperty")
    Set LinkAccess = SolidSection.GetItem("SimLinkAccess")
    LinkAccess.AddLink "MainSupport", Support
  End If
  ' Attributes
    MeshPart.SetAttributeValue "Mesh", "MeshSizeValue", iSize
    MeshPart.SetAttributeValue "Mesh", "ElementOrder", iOrder
    ' If sag requested
    If (iSagValue = "") Then
      MeshPart.SetAttributeValue "Mesh", "AbsoluteSag", 1
    Else
      MeshPart.SetAttributeValue "Mesh", "AbsoluteSag", 2
      MeshPart.SetAttributeValue "Mesh", "AbsoluteSagValue", iSagValue
    End If

End Sub
