' <!-- -------------- Sample macro written in VB ------------------------------------------ --> 
' <!-- This sample is an automated modeling procedure to creates a rule based mid surface Mesh part with abstraction shape. --> 

' <!-- Steps : --> 

' <!-- 1 - Inputs definitions --> 
' <!-- 2 - VB Script to create Mesh Part --> 
	' <!-- a - PLM operations --> 
	' <!-- b - Geometrical operations --> 
	' <!-- c - FE operations --> 

' <!-- -------------------------------------------------------------------------- --> 
' <!-- 1 - Inputs definitions --> 
' <!-- -------------------------------------------------------------------------- --> 

'<Lists>
  '<List Name = "SelectionDefinition">
    '<Member Name = "MainBody" Nls = "MainBody"/>
    '<Member Name = "AllGeometries" Nls = "AllGeometries"/>
    '<Member Name = "VisibleGeometries" Nls = "VisibleGeometries"/>
   '</List>
'</Lists>
'<Inputs>
  '<Input Name = "SelectionMode" Type = "List" SubType = "SelectionDefinition" Mand ="false" Nls ="SelectionMode"/>
  '<Input Name = "MaxRadius" Type ="Double" Mand ="true" Nls ="MaxFilletRadius" Min="0" MinExcl="false"/>
  '<Input Name = "MeshingRule" Type ="DmtDocument" SubType ="CATFmtSurfQuadRule" Mand ="true" Nls ="Rule"/>
  '<Input Name = "Property" Type ="Boolean" Mand ="true" Nls ="CreateShell"/>
  '<Input Name = "PublicationName" Type ="String" Mand ="true" Nls ="PublicationName"/>
'</Inputs>

' <!-- ---------------------------------------------------------------------------------------------------- --> 
' <!-- 2 - VB Script to create Mesh Part --> 
' <!-- ---------------------------------------------------------------------------------------------------- --> 

Sub main(iMode,iMaxRadius,iRule,iProp,iName)

  '
  ' a - PLM operations
  '----------------------------------------------------------------------------------------------------

  'Get Shape Ref and Part
  Dim MyRepRef As VPMRepReference
  Set MyRepRef = Shape.ReferenceInstanceOf
  Dim MyPart As Part
  Set MyPart = MyRepRef.GetItem("Part")
  'Get FEM
  Dim MyFEMRoot As SimFemRoot
  Set MyFEMRoot = FEM.GetItem("SimFemRoot")
  'Get PLM Product Service
  Dim MyProductService As PLMProductService
  Set MyProductService = CATIA.ActiveEditor.GetService("PLMProductService")	  
  'Get SimComposeLinkServices  
  Dim FEMService As SimComposeLinkServices
  Set FEMService = FEM.GetItem("SimComposeLinkServices")

  '------
  ' Build the list of solid bodies for all selection modes
  '------	  
  Dim EmptyBody As Boolean
  Dim count3DGeom As Integer
  count3DGeom = 0
  Dim CheckVisibility As Boolean
  CheckVisibility = False
  If (iMode = "3") Then
    CheckVisibility = True
  End If 
  Dim myList3DGeom
  Dim lShapes(0)
  ' Get AutomatedFEMServices	 	
  Dim AutomatedFEMServices As SimAutoModServices
  Set AutomatedFEMServices = CATIA.ActiveEditor.GetService("SimAutoModServices")  

  '------
  'Selection Mode MainBody
  '------
  If (iMode = "1") Then
    ReDim myList3DGeom(0)
    Dim PartBody As AnyObject
    Set PartBody = MyPart.MainBody
     'Check if PartBody is empty
    EmptyBody = IsAnEmptyBody(PartBody)
    If EmptyBody = True Then
      Set PartBody = FindFirstSolidBody(MyPart)
      If Not (PartBody Is Nothing) Then
        EmptyBody = False
        Set myList3DGeom(0)=PartBody
      End If
    Else
      Set myList3DGeom(0)=PartBody
    End If
    If EmptyBody = True Then
      Exit Sub 
    End If
  End If 

  '------
  'Selection Mode AllGeometries and VisibleGeometries
  '------
  If (iMode = "2") or (iMode = "3") Then
    ' Get all solid results in 3DPart
    Dim myListResults
    On Error Resume Next
    myListResults = AutomatedFEMServices.GetAllResults(MyPart, 3)
    If Err = 0 Then
    	BuildList3DGeom MyPart, myListResults, CheckVisibility, myList3DGeom
    	count3DGeom = UBound(myList3DGeom) - LBound(myList3DGeom) + 1
    End if 
	On Error GoTo 0
    If count3DGeom = 0 Then
      Exit Sub 
    End If
  End If 

  '------
  ' Create and Initialize the Abstraction Shape if not already exists
  '------

  ' Check if publication already exists
  Dim MyAbsShpRef As VPMRepReference
  Dim MyAbsShape As VPMRepInstance
  Dim MyPubAlreadyExists As Boolean
  MyPubAlreadyExists = False
  Dim MyPub As VPMPublication
  Set MyPub = Nothing
  Dim PubName As CATBSTR
  If (iMode = "1") Then
    PubName = iName
  Else
    PubName = iName + ".1"
  End If
  GetPublication MyRepRef, PubName, MyPub, MyPubAlreadyExists

  ' If the Publication already exists we are in MOA case : no need to create the Abs Shape 
  If ((IsEmpty(MOAFEM)) or ( (IsEmpty(MOAFEM)=False) and (MyPubAlreadyExists=False) )) Then
    Set lShapes(0) = MyRepRef
    CreateInitAbsShape Shape, lShapes, myList3DGeom, MyAbsShpRef
    Set MyAbsShape = MyAbsShpRef.ParentRepInstances.Item(1)
  Else
    If MyPubAlreadyExists Then 
      GetAbsShapeRep MyPub, MyAbsShape
      Set MyAbsShpRef = MyAbsShape.ReferenceInstanceOf
      ' Link the FEMRep to the Abstraction Shape
      Set MyLinkToExistingAbstractionShape = FEMService.ComposeLink(OCCURRENCE, Nothing, MyPub)
      MyFEMRoot.AddAssociatedRep MyLinkToExistingAbstractionShape
    End If 
  End If

  ' Get all 3D results of new 3DPart after AbsShape initialization
  Dim MyNewPart As Part
  Set MyNewPart = MyAbsShpRef.GetItem("Part")
  Dim myNewList3DGeom
  myNewList3DGeom = AutomatedFEMServices.GetAllResults(MyNewPart, 3)
  count3DGeom = UBound(myNewList3DGeom) - LBound(myNewList3DGeom) + 1

  ' Loop on 3D results
  For i = 0 To count3DGeom - 1
    ' Get body and reference
    Dim lNewBody As Body
    Set lNewBody = myNewList3DGeom(i)
    Dim BodyRef As Reference
    Set BodyRef = MyNewPart.CreateReferenceFromObject(lNewBody)

    ' Publication name
    Dim iPubCount As Integer
    iPubCount = i + 1
    Dim PubliName As CATBSTR
    If (iMode = "1") Then
      PubliName = iName
    Else
      PubliName = iName + "." + CStr(iPubCount)
    End If
    
    ' MOA case : GetPublication
    Dim MyPublication As VPMPublication
    Set MyPublication = Nothing
    If MyPubAlreadyExists Then
      GetPublication MyRepRef, PubliName, MyPublication, MyPubAlreadyExists
      'Get the link to the Abstraction Shape PartBody for the shell section "secondary support"
      If (iProp = 1) Then
        Set Link2 = MyProductService.ComposeLink(OCCURRENCE, MyAbsShape, BodyRef)
      End If
      
    Else

      '
      ' b - Geometrical operations
      '----------------------------------------------------------------------------------------------------

      '------
      ' Creation feature defeaturing
      '------
      If iMaxRadius > 0 Then 
        MyNewPart.InWorkObject = lNewBody
        CreateDefeaturing MyNewPart, iMaxRadius
      End If 
	 
      '------
      ' Creation midsurface
      '------
      Dim MidSurface As HybridShapeMidSurface
      Set MidSurface = CreateMidSurface(MyNewPart, BodyRef)
		
      '
      ' c - FE operations
      '----------------------------------------------------------------------------------------------------
      'Link the FEMRep to the Abstraction Shape 
      Dim MyReference As Reference
      Set MyReference = MyNewPart.CreateReferenceFromObject(lNewBody)
      Dim myPublicationServices As SimPublicationServices
      Set myPublicationServices = CATIA.ActiveEditor.GetService("SimPublicationServices")
      Set MyLink = MyProductService.ComposeLink(OCCURRENCE, MyAbsShape, MyReference)
      MyFEMRoot.AddAssociatedRep MyLink

      '------
      ' Create a link to the midsurface
      '------
      Dim SupportRef As Reference
      Set SupportRef = MyNewPart.CreateReferenceFromObject(MidSurface)
      Set Link = MyProductService.ComposeLink(OCCURRENCE, MyAbsShape, SupportRef)
      Set Link2 = MyProductService.ComposeLink(OCCURRENCE, MyAbsShape, BodyRef)

      '------
      ' Create a publication on the midsurface
      '------
      Dim myPrdRef As VPMReference
      Set myPrdRef = MyAbsShape.Parent.Parent
      Set MyPublication = myPublicationServices.CreatePublication(myPrdRef, Link, PubliName)
      
    End If
 
    CreateMeshPart MyFEMRoot, MyProductService, MyPublication, Link2, iRule, iProp
    
    'Unlink FEMRep from original 3DShape
    If (MyPubAlreadyExists=False) Then
      Set MyReference = MyPart.CreateReferenceFromObject(MyPart.MainBody)
      Set MyLink = MyProductService.ComposeLink(OCCURRENCE, Shape, MyReference)
      MyFEMRoot.RemoveAssociatedRep MyLink
    End If
			
  Next
 
End Sub

Private Function IsAnEmptyBody(ByVal iBody As Body) As Boolean

If iBody Is Nothing Then
    Exit Function
Else
    Dim myShapes As Shapes
    Set myShapes = iBody.Shapes
    Dim myHybridShapes As HybridShapes
    Set myHybridShapes = iBody.HybridShapes
    IsAnEmptyBody = True
	If Not (myShapes Is Nothing) Then
		If myShapes.Count > 0 Then
			IsAnEmptyBody = False
		End If
	End If
	If Not (myHybridShapes Is Nothing) Then
		If myHybridShapes.Count > 0 Then
			IsAnEmptyBody = False
		End If
	End If
End If

End Function

' Find the fisrt volume in OGS
'----------------------------------------------------------------------------------------------------
Private Function FindFirstSolidBody(ByVal iPart As Part) As Body

  Dim theMeasureService As MeasurableService
  Set theMeasureService = CATIA.ActiveEditor.GetService("MeasurableService")
  found = False
  For Each lOGS In iPart.OrderedGeometricalSets  
    For Each lHybShape In lOGS.HybridShapes
      Dim theMeasurableVolume As MeasurableVolume
      Set theMeasurableVolume = theMeasureService.GetMeasurable(lHybShape, CAAMeasurableVolume)
      Dim theArea As Double
      On Error Resume Next
      theArea = theMeasurableVolume.GetVolume
      If Err = 0 Then
        Set FindFirstSolidBody = lHybShape
        found = True
        Exit For
      End If
      On Error Goto 0
    Next
    If found=True Then Exit For
  Next
   On Error Goto 0
   If found=False Then Set FindFirstSolidBody=Nothing
   
End Function

' Get Publication from its name
'----------------------------------------------------------------------------------------------------
Private Function GetPublication(ByVal MyRepRef As VPMRepReference, ByVal PubliName As String, ByRef MyPublication As VPMPublication, ByRef MyPublicationAlreadyExists As Boolean)

  ' Retrieve the publication if it exists
  Dim MyRef As VPMReference
  Set MyRef = MyRepRef.Father

  Dim ListPublications As VPMPublications
  Set ListPublications = MyRef.Publications

  On Error Resume Next
  Set MyPublication = ListPublications.GetItem (PubliName)
  On Error Goto 0
	
  MyPublicationAlreadyExists = Not (MyPublication Is Nothing)
	
End Function

' Get Abstraction Shape from a Publication
'----------------------------------------------------------------------------------------------------
Private Sub GetAbsShapeRep(ByVal MyPublication As VPMPublication, ByRef MyPubAbsShape As VPMRepInstance)

  Dim PublicationServices As SimPublicationServices
  Set PublicationServices = CATIA.ActiveEditor.GetService("SimPublicationServices")
  Dim PublishedElement As AnyObject
  PublicationServices.GetPublishedElement MyPublication, PublishedElement
				
  Dim MySimLinkSrv As SimLinkServices
  Set MySimLinkSrv = CATIA.ActiveEditor.GetService("SimLinkServices")
		  
  Set MyPubAbsShape = MySimLinkSrv.GetRepInstance(PublishedElement)

End Sub

' Create and Initialize an Abstraction Shape from list of shapes and features
'----------------------------------------------------------------------------------------------------
Private Sub CreateInitAbsShape(ByVal Shape As AnyObject, ByVal lShapes As AnyObject, ByVal lFeats As AnyObject, ByRef MyAbsShpRef As VPMRepReference)

  ' Create the Abstraction Shape
  Dim MyPrdRepFactory As SimPrdRepFactory
  Set MyPrdRepFactory = Shape.ReferenceInstanceOf.Father.GetItem("SimPrdRepFactory")
  Set MyAbsShpRef = MyPrdRepFactory.CreatePrdRep("AbstractionShape")
  Dim MyAbsShape As VPMRepInstance
  Set MyAbsShape = MyAbsShpRef.ParentRepInstances.Item(1)

  ' Initialize the Abstraction Shape
  InitAbsShape MyAbsShpRef, lShapes, lFeats
	
End Sub

' Initialize a given Abstraction Shape from list of shapes and features
'----------------------------------------------------------------------------------------------------
Private Sub InitAbsShape(ByVal MyAbsShpRef As VPMRepReference, ByVal lShapes As AnyObject, ByVal lFeats As AnyObject)

  Dim AshInit As SimRepInitialization
  Set AshInit = MyAbsShpRef.GetItem("SimRepInitialization")
  Dim InitParams 'As SimParameterSet
  Set InitParams = AshInit.Parameters
  InitParams.SetStringParameter "Mode", "LocalImport"
  InitParams.SetObjectArrayParameter "3DShapes", lShapes
  InitParams.SetObjectArrayParameter "FeaturesToCopy", lFeats
  InitParams.SetIntegerParameter "EraseBeforeCopy", 1
  InitParams.SetIntegerParameter "Assemble3DBodies", 0
  AshInit.InitializeRep (InitParams)

End Sub

' Build list of 3D results according to visibility flag
'----------------------------------------------------------------------------------------------------
Private Sub BuildList3DGeom(ByVal MyPart As Part, ByVal myListResults As AnyObject, ByVal CheckVisibility As Boolean, ByRef myList3DGeom As AnyObject)
  
  Dim countResults As Integer
  countResults = UBound(myListResults) - LBound(myListResults) + 1
  Dim count3DGeom As Integer
  count3DGeom = 0
  ReDim myList3DGeom(countResults)
  
  'Loop on Results
  For i = 0 To countResults - 1
    Dim Support As Body 
    Set Support = myListResults(i)
    Dim Mysel As Selection
    Set Mysel = CATIA.ActiveEditor.Selection
    Dim visStatus As CatVisPropertyShow
    Mysel.Add Support
    Mysel.VisProperties.GetShow visStatus
    Mysel.Clear
    If (CheckVisibility = False) or ((CheckVisibility = True) and (visStatus = catVisPropertyShowAttr)) Then
      Set myList3DGeom(count3DGeom) = Support
      count3DGeom = count3DGeom + 1
    End If 
  Next 
  'Resize arrays
  count3DGeom = count3DGeom - 1
  ReDim Preserve myList3DGeom(count3DGeom)
	
End Sub

' Create Defeaturing
'----------------------------------------------------------------------------------------------------
Private Sub CreateDefeaturing(ByVal MyNewPart As Part, iMaxRadius)

  Dim myShapeFactory As ShapeFactory
  Set myShapeFactory = MyNewPart.ShapeFactory
  Dim defeature As Defeaturing
  Set defeature = myShapeFactory.AddNewDefeaturing()
  
  ''''' Retrieval of the filter collection associated to the created defeaturing object
  Dim Filters As DefeaturingFilters
  Set Filters = defeature.Filters
  
  ''''' Creation and retrieval filletfilter
  filletFilterIndex = Filters.Add("DefeaturingFilletFilter")
  Dim filletFilter As DefeaturingFilletFilter
  Set filletFilter = Filters.Item(filletFilterIndex)
  
  ''''' Customize fillet filter
  filletFilter.setMaximumValue "RibbonRadius", iMaxRadius

End Sub

' Create MidSurface
'----------------------------------------------------------------------------------------------------
Private Function CreateMidSurface( ByVal MyNewPart as Part, ByVal BodyRef as Reference ) As HybridShapeMidSurface

  ''''' Create a new Geometrical Set
  Dim NewGeomSet As HybridBody
  If MyNewPart.HybridBodies.Count > 0 Then
  	Set NewGeomSet = MyNewPart.HybridBodies.Item(1)
  Else
    Set NewGeomSet = MyNewPart.HybridBodies.Add
  End If
  
  NewGeomSet.Name = "midsurface"
  
  ''''' Create a midsurface
  Dim hybridShapeFactory As hybridShapeFactory
  Set hybridShapeFactory = MyNewPart.hybridShapeFactory
  
  Set CreateMidSurface = hybridShapeFactory.AddNewMidSurfaceWithAutoThreshold(BodyRef, 2, 7, 1)
  
  ''''' Append the midsurface to the Geometrical Set
  NewGeomSet.AppendHybridShape CreateMidSurface
  
  ''''' Update
  hybridShapeFactory.GSMVisibility CreateMidSurface, 1
  hybridShapeFactory.GSMVisibility BodyRef, 0
  
End Function

' Create Mesh Part
'----------------------------------------------------------------------------------------------------
Private Sub CreateMeshPart(ByVal MyFEMRoot As CATIAFmtFemRoot, ByVal MyProductService As PLMProductService, ByVal MyPublication As VPMPublication, ByVal Link2 As AnyObject, ByVal iRule As AnyBody, ByVal iProp As Integer)

  ''''' Create the surface mesh part
  Dim MyMeshSet As SimMeshSet
  Set MyMeshSet = MyFEMRoot.GetSet("SimNodesElements")
  Dim MyMeshParts As SimMeshParts
  Set MyMeshParts = MyMeshSet.MeshParts
  Dim MyMeshPart As SimMeshPart
  Set MyMeshPart = MyMeshParts.Add("CATFmtSurfQuadMesher")
  
  ''''' Set the publication as support
  Set MyDeathLink = MyProductService.ComposeLink(OCCURRENCE, Nothing, MyPublication)
  MyMeshPart.AddSupport(MyDeathLink)
  
  ''''' Set the meshing rule
  MyMeshPart.SetRule iRule,"CATFmtSurfQuadRule"
  
  ''''' Create the shell property
  If (iProp = 1) Then
    Dim PropertySet As SimProperties
    Set PropertySet = MyFEMRoot.GetSet("SimProperties")
    Dim ShellSection As SimShellSection
    Set ShellSection = PropertySet.Add("SimShellSection")
    ShellSection.ThicknessType = SimShellSectionMid
    
    Dim LinkAccess As SimLinkAccess
    Set LinkAccess = ShellSection.GetItem("SimLinkAccess")
    If IsEmpty(MOAFEM) Then
      LinkAccess.AddLink "MainSupport", MyPublication
      LinkAccess.AddLink "SecondarySupport", Link2
    Else
      LinkAccess.AddLinkWithContext "MainSupport", MyPublication, OCCURRENCE
      LinkAccess.AddLinkWithContext "SecondarySupport", Link2, OCCURRENCE
    End If
  End If

End Sub
