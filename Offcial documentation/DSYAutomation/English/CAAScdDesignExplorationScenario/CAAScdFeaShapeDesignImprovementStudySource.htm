'-------------- Sample MACRO written in Basic Script ------------------------------------------
'This sample creates and executes a Shape Design Improvement Study.

'Steps:
'          1- Retrieving the active Product 										
'          2- Retrieving the Publications and Composing Links 								
'          3- Creating the FEM Rep and retrieving its Rep Manager 		
'          4- Creating a Mesh on a Part								
'          5- Creating a Property 					
'          6- Initializing and Creating the Simulation PLM Object						
'          7- Retrieving the Scenario Manager							
'          8- Creating the Structural Analysis Case with the FEM rep
'          9- Creating and initializing the static step				
'          10- Retrieving the structural analysis case features and features states sets		
'          11- Creating the structural analysis case features								
'          12- Creating the Shape Design Improvement Study with the Structural Analysis Case
'          13- Retrieving the Shape Design Improvement Study  features
'          14- Creating the Shape Design Improvement Study  features
'          15- Launching the Shape Design Improvement Study									
'============================================================================================

Sub CATMain()
	
'=====================================================================
' 1- Retrieving the active Product
'=====================================================================
    
	Dim myEditor As Editor
	Set myEditor = CATIA.ActiveEditor
	
	Dim myProdService As PLMProductService
	Set myProdService = myEditor.GetService("PLMProductService")

	Dim myRootOccurrence As VPMRootOccurrence
	Set myRootOccurrence= myProdService.RootOccurrence
	
'Retrieving the reference of root occurence
'------------------------------------------
	
	Dim myModel As VPMReference
	Set myModel= myRootOccurrence.ReferenceRootOccurrenceOf
	
	
'=====================================================================
' 2- Retrieving the Publications and Composing Links
'=====================================================================
	
	Dim myPublications  As VPMPublications
	Set myPublications  = myModel.Publications
	
	Dim myLoadSupport as VPMPublication
	Set myLoadSupport= myPublications.GetItem("Loading_Support")
	
	Dim myClampSupport1 as VPMPublication
	Set myClampSupport1 = myPublications.GetItem("Clamp_Support_1")

	Dim myClampSupport2 as VPMPublication
	Set myClampSupport2 = myPublications.GetItem("Clamp_Support_2")
	
	Dim myDesignSpaceSupport as VPMPublication
	Set myDesignSpaceSupport= myPublications.GetItem("Design_Space_Support")
	
	Dim myDisconnectedAreaSupport as VPMPublication
	Set myDisconnectedAreaSupport= myPublications.GetItem("Local_Opt_Area_Support")
	
'Composing links
'---------------
	
	Dim myRepInstance As VPMRepInstance
	Set myRepInstance = Nothing ' Because we use publications
	
	Dim myLinkToLoadSupport as AnyObject
	Set myLinkToLoadSupport = myProdService.ComposeLink(myRootOccurrence, myRepInstance, myLoadSupport)
	
	Dim myLinkToClampSupport1 as AnyObject
	Set myLinkToClampSupport1 = myProdService.ComposeLink(myRootOccurrence, myRepInstance, myClampSupport1)

	Dim myLinkToClampSupport2 as AnyObject
	Set myLinkToClampSupport2 = myProdService.ComposeLink(myRootOccurrence, myRepInstance, myClampSupport2)
	
	Dim myLinkToDisconnectSupport as AnyObject
	Set myLinkToDisconnectSupport = myProdService.ComposeLink(myRootOccurrence, myRepInstance, myDesignSpaceSupport)
	
	Dim myLinkToDisconnectedArea as AnyObject
	Set myLinkToDisconnectedArea= myProdService.ComposeLink(myRootOccurrence, myRepInstance, myDisconnectedAreaSupport)	
	
'=====================================================================
' 3- Creating the FEM Rep and retrieving its Rep Manager
'=====================================================================
	
	Dim myPrdRepFactory as SimPrdRepFactory
	Set myPrdRepFactory =myModel.GetItem("SimPrdRepFactory")
	
	Dim myRepRef As VPMRepReference
	Set myRepRef = myPrdRepFactory.CreatePrdRep("FEM")

	Dim myRepInstances As VPMRepInstances
	Set myRepInstances= myModel.RepInstances
	
	Dim myFEMRoot As SimFemRoot

 For i = 1 To myRepInstances.Count
 Dim oRepInstance As VPMRepInstance
 Set oRepInstance = myRepInstances.Item(i)

 Dim oRepReference As VPMRepReference
 Set oRepReference = oRepInstance.ReferenceInstanceOf

 If oRepReference.GetAttributeValue("V_discipline") = "FEM" Then
 Set myRepRef  = oRepReference
 Set myFEMRoot = oRepReference.GetItem("SimFemRoot")
 End If 
 Next 
	
         myFEMRoot.AddAssociatedRep myDesignSpaceSupport
  
'=====================================================================
' 4- Creating a Mesh on a Part
'=====================================================================
	
	Dim myMeshSet As SimMeshSet
	Set myMeshSet = myFEMRoot.GetSet("SimNodesElements")
	
	Dim myMeshParts As SimMeshParts
	Set myMeshParts = myMeshSet.MeshParts

	Dim myMeshPart As SimMeshPart
	Set myMeshPart = myMeshParts.Add("CATFmt3DRulesMesher")

	myMeshPart.AddSupport myDesignSpaceSupport
	
	myMeshPart.SetAttributeValue "Mesh", "MeshSizeValue", "3.3mm"
	myMeshPart.SetAttributeValue "Mesh", "ElementOrder", "1"	
	myMeshSet.Update
	
'=====================================================================
' 5- Creating a Property
'=====================================================================
	
	Dim myPropertiesSet as SimProperties
	Set myPropertiesSet = myFEMRoot.GetSet("SimProperties")

	Dim mySolidSection As SimSolidSection
	Set mySolidSection = myPropertiesSet.Add("SimSolidSection")

	Dim myLinkAccess As SimLinkAccess
	Set myLinkAccess = mySolidSection.GetItem("SimLinkAccess")
	myLinkAccess.AddLink "MainSupport", myLinkToDisconnectSupport 
	
'=====================================================================
' 6- Initializing and Creating the Simulation PLM Object
'=====================================================================
	
	Dim myInitService As SimInitializationService
	Set myInitService = CATIA.GetSessionService("SimInitializationService")
	
	myInitService.SetSimulationInitialization "SimDesignExplorationCreation", "SimOptimization"

	Dim mySimPLMService as SIMPLMService
	Set mySimPLMService = CATIA.GetSessionService("SIMPLMService")
	
	Dim mySimulationReference As SimulationReference
	mySimPLMService.PLMCreate "NewShapeDesignImprovementStudy", "SMAFeaPLMNewSimu", myModel, mySimulationReference, myEditor
	
'=====================================================================
' 7- Retrieving the Scenario Manager
'=====================================================================
    
	Dim myScenarioManager As SimScenarioManager
	Set myScenarioManager = mySimulationReference.GetItem("SimScenarioManager")

'=====================================================================
' 8- Creating the Structural Analysis Case with the FEM rep
'=====================================================================
    
	Dim myAnalysisCases As SimAnalysisCases
	Set myAnalysisCases = myScenarioManager.AnalysisCases
	
	Dim myAnalysisCase As SimAnalysisCase
	Set myAnalysisCase = myAnalysisCases.AddAnalysisCase("SimStructuralAnalysisCase")	
	myAnalysisCase.FEMRep = myFEMRoot 
	
'=====================================================================
' 9- Creating and initializing the static step
'=====================================================================
    
	Dim mySteps As SimSteps
	Set mySteps = myAnalysisCase.Steps
	
	Dim myStaticStep As SimStaticStep
	Set myStaticStep = mySteps.Add("SimStaticStep")
	
	myAnalysisCase.CreateDefaultElementTypeAssignment
	
	myStaticStep.CreateAnalysisDefaultOutputRequests
	myStaticStep.MaximumIncrements = 1000

'=====================================================================
' 10- Retrieving the features and features states sets
'=====================================================================
    
	Dim myAnalysisCaseFeatures As SimFeatures
	Set myAnalysisCaseFeatures = myAnalysisCase.Features

	Dim myAnalysisCaseFeatureStates As SimFeatureStates
	Set myAnalysisCaseFeatureStates = myStaticStep.FeatureStates

'=====================================================================
' 11- Creating features
'=====================================================================
	
' Creating a clamp feature
'------------------------------------------
	
	Dim myClamp1 As SimClamp
	Set myClamp1 = myAnalysisCaseFeatures.Add("SimClamp")
	
	Set myLinkAccess = myClamp1.GetItem("SimLinkAccess")
	myLinkAccess.AddLink "MainSupport", myLinkToClampSupport1
	
	myAnalysisCaseFeatureStates.CreateFeatureState myClamp1

	Dim myClamp2 As SimClamp
	Set myClamp2 = myAnalysisCaseFeatures.Add("SimClamp")
	
	Set myLinkAccess = myClamp2.GetItem("SimLinkAccess")
	myLinkAccess.AddLink "MainSupport", myLinkToClampSupport2
	
	myAnalysisCaseFeatureStates.CreateFeatureState myClamp1
	
' Creating a force feature
'----------------------------
	
	Dim myLoad As SimForce
	Set myLoad= myAnalysisCaseFeatures.Add("SimForce")
	
	Set myLinkAccess = myLoad.GetItem("SimLinkAccess")
	myLinkAccess.AddLink "MainSupport", myLinkToLoadSupport	
	myLoad.SetMagnitude 0,0,-100
	
	myAnalysisCaseFeatureStates.CreateFeatureState myLoad
	
'================================================================================================
'12- Creating the Shape Design Improvement Study with the structural analysis case
'================================================================================================
    	
	Dim myShapeStudy As SimShapeDesignImprovementStudy
	Set myShapeStudy= myAnalysisCases.AddAnalysisCase("SimShapeDesignImprovementStudy")	
	myShapeStudy.AnalysisCase = myAnalysisCase

'=====================================================================
' 13- Retrieving the Shape design improvement study features
'=====================================================================
    
	Dim myShapeFeatures As SimDesignImprovementFeatures
	Set myShapeFeatures= myShapeStudy.Features
	
	
'=====================================================================
' 14- Creating Shape design improvement study features
'=====================================================================
	
' Creating a design space feature
'-------------------------
	
	Dim myDesignSpace As SimDesignEnvelope
	Set myDesignSpace= myShapeFeatures.Add("SimDesignEnvelope")
	
	Set myLinkAccess =myDesignSpace.GetItem("SimLinkAccess")
	myLinkAccess.AddLink "MainSupport", myDesignSpaceSupport

' Creating a local optimization area feature
'-------------------------
	
	Dim myLocalOptimizationArea As SimLocalOptimizationArea
	Set myLocalOptimizationArea= myShapeFeatures.AddSubFeature("SimLocalOptimizationArea", myDesignSpace)
	
	Set myLinkAccess =myLocalOptimizationArea.GetItem("SimLinkAccess")
	myLinkAccess.AddLink "MainSupport", myLinkToDisconnectedArea
	myLocalOptimizationArea.SetTransitionDistance (0.005)


' Creating a nonparametric objective feature
'-------------------------
	
	Dim myObjective As SimNonParametricObjective
	Set myObjective= myShapeFeatures.Add("SimNonParametricObjective")
	
	myObjective.Type = SimMinimizeMaximumStress
	myObjective.TargetMassType = SimRatio
	myObjective.SetTargetMassRatio (0.99)

'=====================================================================
' 15- Launching the shape design improvement study 
'=====================================================================
    
	Dim myExecutionService As SimExecutionService
	Set myExecutionService = myEditor.GetService("SimExecutionService")
	
	myExecutionService.BasicExecuteAll mySimulationReference
	
' Switching to the results visualization workbench
'-------------------------------------------------
	
	CATIA.StartWorkbench "SMAHvcStructureResultsWorkbench"

End Sub
