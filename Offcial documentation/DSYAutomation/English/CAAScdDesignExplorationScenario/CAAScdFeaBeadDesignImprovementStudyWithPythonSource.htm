# -------------- Sample script written in Python ------------------------------------------
# This sample initializes and executes a Bead design improvement study.

# Steps:
#          1  - Retrieves the active Product 
#          2  - Retrieves the Publications and Composes Links
#          3  - Creates the FEM Rep and retrieves its Rep Manager
#          4  - Creates a Mesh on a Part
#          5  - Creates a Property
#          6  - Initializes and creates the Simulation PLM Object
#          7  - Retrieves the Scenario Manager
#          8  - Creates the Structural Analysis Case with the FEM rep
#          9  - Creates and initializes the static step
#          10 - Retrieves the structural analysis case features and features states sets
#          11 - Creates the structural analysis case features
#          12 - Creates the Bead Design Improvement Study with the structural analysis case	
#          13 - Retrieves the Bead design improvement study  features
#          14 - Creates the Bead design Improvement study  features
#          15 - Launches the Bead design improvement study                                                                              
# ============================================================================================

import sys, os
from com3dx import *
CATIA = get3dxClient()

#--------------------------------------------------------------------------
# 1 - Retrieves the active Product
#--------------------------------------------------------------------------

myEditor = CATIA.ActiveEditor
myProdService = myEditor.GetService("PLMProductService")
myRootOccurrence = myProdService.RootOccurrence
myModel= myRootOccurrence.ReferenceRootOccurrenceOf

#--------------------------------------------------------------------------
# 2 - Retrieves the Publications and Composes Links
#--------------------------------------------------------------------------
myPublications  = myModel.Publications

myLoadSupport= myPublications.GetItem("Loading_Support")
myClampSupport1 = myPublications.GetItem("Clamp_Support1")
myClampSupport2 = myPublications.GetItem("Clamp_Support2")
myDesignSpaceSupport= myPublications.GetItem("Design_Space_Support")
myDisconnectedAreaSupport= myPublications.GetItem("Local_Opt_Area_Support")



# Composing links
# ---------------
myLinkToLoadSupport = myProdService.ComposeLink(myRootOccurrence, Nothing, myLoadSupport)
myLinkToClampSupport1 = myProdService.ComposeLink(myRootOccurrence, Nothing, myClampSupport1)
myLinkToClampSupport2 = myProdService.ComposeLink(myRootOccurrence, Nothing, myClampSupport2)
myLinkToDisconnectSupport = myProdService.ComposeLink(myRootOccurrence, Nothing, myDesignSpaceSupport)
myLinkToDisconnectedArea = myProdService.ComposeLink(myRootOccurrence, Nothing, myDisconnectedAreaSupport)


#--------------------------------------------------------------------------
# 3 - Creates the FEM Rep and retrieves its Rep Manager
#--------------------------------------------------------------------------
myPrdRepFactory =myModel.GetItem("SimPrdRepFactory")
myRepRef = myPrdRepFactory.CreatePrdRep("FEM")
myRepInstances= myModel.RepInstances

For i = 1 To myRepInstances.Count
    	 oRepInstance = myRepInstances.Item(i)
    	 oRepReference = oRepInstance.ReferenceInstanceOf
    	 If oRepReference.GetAttributeValue("V_discipline") = "FEM" Then
    	    myRepRef  = oRepReference
    	    myFEMRoot = oRepReference.GetItem("SimFemRoot")
    	 End If 
Next 
	
myFEMRoot.AddAssociatedRep (myDesignSpaceSupport)

#--------------------------------------------------------------------------
# 4 - Creates a Mesh on a Part
#--------------------------------------------------------------------------
myMeshSet = myFEMRoot.GetSet("SimNodesElements")
myMeshParts = myMeshSet.MeshParts
myMeshPart = myMeshParts.Add("CATFmt3DRulesMesher")
myMeshPart.AddSupport (myDesignSpaceSupport)

myMeshPart.SetAttributeValue ("Mesh", "MeshSizeValue", "3.3mm")
myMeshPart.SetAttributeValue ("Mesh", "ElementOrder", "1.0")	

myMeshSpecifications = myMeshPart.GetTopologySpecifications()
myMeshSpecification = myMeshSpecifications.Add("CATFmtConstrainEdgeLocalSpec")
	
myLinkAccess = MyMeshSpecification.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myLinkToDisconnectSupport

myMeshSet.Update()

#--------------------------------------------------------------------------
# 5 - Creates a Property
#--------------------------------------------------------------------------

myPropertiesSet = myFEMRoot.GetSet("SimProperties")
myShellSection = myPropertiesSet.Add("SimShellSection")

myLinkAccess = myShellSection.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToDisconnectSupport)

#--------------------------------------------------------------------------
# 6 - Retrieves the Scenario Manager
#--------------------------------------------------------------------------
myScenarioManager = MySimulationRoot.GetItem("SimScenarioManager")

#--------------------------------------------------------------------------
# 7 - Associates the Structural Analysis Case with the FEM rep
#--------------------------------------------------------------------------
myAnalysisCases = myScenarioManager.AnalysisCases
myAnalysisCase = myAnalysisCases.AddAnalysisCase("SimStructuralAnalysisCase")
myAnalysisCase.FEMRep = myFEMRoot 

#--------------------------------------------------------------------------
# 8 - Creates and initializes the static step
#--------------------------------------------------------------------------
mySteps = myAnalysisCase.Steps
myStaticStep = mySteps.Add("SimStaticStep")

myAnalysisCase.CreateDefaultElementTypeAssignment()
myStaticStep.CreateAnalysisDefaultOutputRequests()

myStaticStep.MaximumIncrements = 1000

#--------------------------------------------------------------------------
# 9 - Retrieves the features and features states sets
#--------------------------------------------------------------------------
myAnalysisCaseFeatures = myAnalysisCase.Features
myAnalysisCaseFeatureStates = myStaticStep.FeatureStates

#--------------------------------------------------------------------------
# 10 - Creates features
#--------------------------------------------------------------------------

# Creates a Clamp feature
# -----------------------------------------
myClamp1 = myAnalysisCaseFeatures .Add("SimClamp")
myLinkAccess = myClamp1.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToClampSupport1)

myAnalysisCaseFeatureStates.CreateFeatureState (myClamp1)

myClamp2 = myAnalysisCaseFeatures .Add("SimClamp")
myLinkAccess = myClamp2.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToClampSupport2)

myAnalysisCaseFeatureStates.CreateFeatureState (myClamp2)

# Creates a force feature
# ------------------------------
myLoad= myAnalysisCaseFeatures.Add("SimForce")
myLinkAccess = myLoad.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToLoadSupport)
myLoad.SetMagnitude 0,0,-100

myAnalysisCaseFeatureStates.CreateFeatureState myLoad

# Creates the Bead Design Improvement Study with the structural analysis case 
#--------------------------------------------------------------------------------
myBeadStudy= myAnalysisCases.AddAnalysisCase("SimBeadDesignImprovementStudy")
myBeadStudy.AnalysisCase = myAnalysisCase

#--------------------------------------------------------------------------------
# Creates the Bead Design Improvement Study Features 
#--------------------------------------------------------------------------------
myBeadFeatures= myBeadStudy.Features

#-------------------------------------------------------------------------------	
# Creating Bead design improvement study features 
#--------------------------------------------------------------------------------
	
# Creating a design space feature
#--------------------------------
myDesignSpace= myBeadFeatures.Add("SimDesignEnvelope")
myDesignSpace.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myLinkToDisconnectSupport 

# Creating a preserved region feature
#------------------------------------

myLocalOptimizationArea= myBeadFeatures.AddSubFeature("SimLocalOptimizationArea", myDesignSpace)
myLinkAccess = myLocalOptimizationArea.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myLinkToDisconnectedArea 

# Creating a nonparametric objective feature
#-------------------------------------------

myObjective= myBeadFeatures.Add("SimNonParametricObjective")
myObjective.Type = SimMaximizeStiffness

#--------------------------------------------------------------------------
# 11 - Launches the simulation
#--------------------------------------------------------------------------
myExecutionService = myEditor.GetService("SimExecutionService")
myExecutionService.BasicExecuteAll (MySimulationRoot)


# Switching to the results visualization workbench
# ------------------------------------------------
CATIA.StartWorkbench ("SMAHvcStructureResultsWorkbench")
