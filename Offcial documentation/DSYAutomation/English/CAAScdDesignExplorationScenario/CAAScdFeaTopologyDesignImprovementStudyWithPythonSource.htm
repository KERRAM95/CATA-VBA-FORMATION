# -------------- Sample script written in Python ------------------------------------------
# This sample initializes and executes a topology design improvement study.

# Steps:
#          1  - Retrieves the active Product 
#          2  - Retrieves the Publications and Composes Links
#          3  - Creates the FEM Rep and retrieves its Rep Manager
#          4  - Creates a Mesh on a Part
#          5  - Creates a Property
#          6  - Initializes and creates the Simulation PLM Object
#          7  - Retrieves the Scenario Manager
#          8  - Creates the Structural Analysis Case with the FEM rep
#          9  - Creates and initializes the static step
#          10 - Retrieves the structural analysis case features and features states sets
#          11 - Creates the structural analysis case features
#          12 - Creates the Topology Design Improvement Study with the structural analysis case	
#          13 - Retrieves the topology design improvement study  features
#          14 - Creates the topology design Improvement study  features
#          15 - Launches the topology design improvement study                                                                              
# ============================================================================================

import sys, os
from com3dx import *
CATIA = get3dxClient()

#--------------------------------------------------------------------------
# 1 - Retrieves the active Product
#--------------------------------------------------------------------------

myEditor = CATIA.ActiveEditor
myProdService = myEditor.GetService("PLMProductService")
myRootOccurrence = myProdService.RootOccurrence
myModel= myRootOccurrence.ReferenceRootOccurrenceOf

#--------------------------------------------------------------------------
# 2 - Retrieves the Publications and Composes Links
#--------------------------------------------------------------------------
myPublications  = myModel.Publications

myLoadSupport= myPublications.GetItem("Loading_Support")
myClampSupport = myPublications.GetItem("Clamp_Support")
myDesignSpaceSupport= myPublications.GetItem("Design_Space_Support")
myPreservedRegion1= myPublications.GetItem("Preserved_Region_Support_1")
myPreservedRegion2= myPublications.GetItem("Preserved_Region_Support_2")
myPlane = myPublications.GetItem("Symmetry_Control_Support")

# Composing links
# ---------------
myProdService.ComposeLink(myRootOccurrence, Nothing, myLoadSupport)
myProdService.ComposeLink(myRootOccurrence, Nothing, myClampSupport)
myProdService.ComposeLink(myRootOccurrence, Nothing, myDesignSpaceSupport)
myProdService.ComposeLink(myRootOccurrence, Nothing, myPreservedRegion1)
myProdService.ComposeLink(myRootOccurrence, Nothing, myPreservedRegion2)
myProdService.ComposeLink(myRootOccurrence, Nothing, myPlane)

#--------------------------------------------------------------------------
# 3 - Creates the FEM Rep and retrieves its Rep Manager
#--------------------------------------------------------------------------
myPrdRepFactory =myModel.GetItem("SimPrdRepFactory")
myRepRef = myPrdRepFactory.CreatePrdRep("FEM")
myRepInstances= myModel.RepInstances

For i = 1 To myRepInstances.Count
    	 oRepInstance = myRepInstances.Item(i)
    	 oRepReference = oRepInstance.ReferenceInstanceOf
    	 If oRepReference.GetAttributeValue("V_discipline") = "FEM" Then
    	    myRepRef  = oRepReference
    	    myFEMRoot = oRepReference.GetItem("SimFemRoot")
    	 End If 
Next 
	
myFEMRoot.AddAssociatedRep (myDesignSpaceSupport)

#--------------------------------------------------------------------------
# 4 - Creates a Mesh on a Part
#--------------------------------------------------------------------------
myMeshSet = myFEMRoot.GetSet("SimNodesElements")
myMeshParts = myMeshSet.MeshParts
myMeshPart = myMeshParts.Add("CATFmt3DRulesMesher")
myMeshPart.AddSupport (myDesignSpaceSupport)

myMeshPart.SetAttributeValue ("Mesh", "MeshSizeValue", "3.3mm")
myMeshPart.SetAttributeValue ("Mesh", "ElementOrder", "1.0")	
myMeshSet.Update()

#--------------------------------------------------------------------------
# 5 - Creates a Property
#--------------------------------------------------------------------------

myPropertiesSet = myFEMRoot.GetSet("SimProperties")
mySolidSection = myPropertiesSet.Add("SimSolidSection")

myLinkAccess = mySolidSection.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToSupport)

#--------------------------------------------------------------------------
# 6 - Retrieves the Scenario Manager
#--------------------------------------------------------------------------
myScenarioManager = MySimulationRoot.GetItem("SimScenarioManager")

#--------------------------------------------------------------------------
# 7 - Associates the Structural Analysis Case with the FEM rep
#--------------------------------------------------------------------------
myAnalysisCases = myScenarioManager.AnalysisCases
myAnalysisCase = myAnalysisCases.AddAnalysisCase("SimStructuralAnalysisCase")
myAnalysisCase.FEMRep = myFEMRoot 

#--------------------------------------------------------------------------
# 8 - Creates and initializes the static step
#--------------------------------------------------------------------------
mySteps = myAnalysisCase.Steps
myStaticStep = mySteps.Add("SimStaticStep")

myAnalysisCase.CreateDefaultElementTypeAssignment()
myStaticStep.CreateAnalysisDefaultOutputRequests()

myStaticStep.MaximumIncrements = 1000

#--------------------------------------------------------------------------
# 9 - Retrieves the features and features states sets
#--------------------------------------------------------------------------
myAnalysisCaseFeatures = myAnalysisCase.Features
myAnalysisCaseFeatureStates = myStaticStep.FeatureStates

#--------------------------------------------------------------------------
# 10 - Creates features
#--------------------------------------------------------------------------

# Creates a displacement restraint feature
# -----------------------------------------
myDisplacementRestraint = myAnalysisCaseFeatures .Add("SimDisplacementRestraint")
myLinkAccess = myDisplacementRestraint.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToClampSupport)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimTranslation1, True)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimTranslation2, True)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimTranslation3, True)

myAnalysisCaseFeatureStates.CreateFeatureState (myDisplacementRestraint)

# Creates a bearing load feature
# ------------------------------
myBearingLoad= myAnalysisCaseFeatures.Add("SimBearingLoad")
myLinkAccess = myBearingLoad.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToLoadSupport)
myBearingLoad.XMagnitude = 40980
myBearingLoad.YMagnitude = 40980

myAnalysisCaseFeatureStates.CreateFeatureState myBearingLoad

#-------------------------------------------------------------------------------
# Creates the Topology Design Improvement Study with the structural analysis case 
#--------------------------------------------------------------------------------
myTopologyStudy= myAnalysisCases.AddAnalysisCase("SimTopologyDesignImprovementStudy")
myTopologyStudy.AnalysisCase = myAnalysisCase

#--------------------------------------------------------------------------------
# Creates the Topology Design Improvement Study Features 
#--------------------------------------------------------------------------------
myTopologyFeatures= myTopologyStudy.Features

#-------------------------------------------------------------------------------	
# Creating topology design improvement study features 
#--------------------------------------------------------------------------------
	
# Creating a design space feature
#--------------------------------
myDesignSpace= myTopologyFeatures.Add("SimDesignEnvelope")
myDesignSpace.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myLinkToPartitionSupport

# Creating a preserved region feature
#------------------------------------

myPreservedRegion1= myTopologyFeatures.AddSubFeature("SimPreservedRegion", myDesignSpace)
myLinkAccess =myPreservedRegion1.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myLinkToPreservedRegionSupport1

myPreservedRegion2 As SimPreservedRegion
myPreservedRegion2 = myTopologyFeatures.AddSubFeature("SimPreservedRegion", myDesignSpace)
myLinkAccess = myPreservedRegion2.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myLinkToPreservedRegionSupport2


# Creating a nonparametric objective feature
#-------------------------------------------

myObjective= myTopologyFeatures.Add("SimNonParametricObjective")
myObjective.Type = SimMaximizeStiffness
myObjective.TargetMassType = SimRatio
myObjective.SetTargetMassRatio (0.60)


# Creating a symmetry control feature
#------------------------------------

mySymmetryControl= myTopologyFeatures.Add("SimSymmetryControl")
myLinkAccess =mySymmetryControl.GetItem("SimLinkAccess")
myLinkAccess.AddLink "MainSupport", myDesignSpace
myLinkAccess = mySymmetryControl.Plane
myLinkAccess.AddLink "MainSupport", myLinkToSymmetryPlaneSupport

#--------------------------------------------------------------------------
# 11 - Launches the simulation
#--------------------------------------------------------------------------
myExecutionService = myEditor.GetService("SimExecutionService")
myExecutionService.BasicExecuteAll (MySimulationRoot)


# Switching to the results visualization workbench
# ------------------------------------------------
CATIA.StartWorkbench ("SMAHvcStructureResultsWorkbench")
