# -------------- Sample script written in Python ------------------------------------------
# This sample creates and executes a thermal structural simulation.

#Steps:
#          1  - Retrieves the simulation and its product										
#          2  - Retrieves the Publications and Composes Links 								
#          3  - Creates the FEM Rep and retrieves its Rep Manager 		
#          4  - Creates a Mesh on a Part								
#          5  - Creates a Property					
#          6  - Retrieves the Scenario Manager				
#          7  - Associates the Structural Analysis Case with the FEM rep					
#          8  - Creates and initializes the static step					
#          9  - Retrieves the thermal features and features states sets							
#          10 - Creates thermal features
#          11 - Creates and initializes the static step		
#          12 - Retrieves the structural features and features states sets									
#          13 - Creates features		
#          14 - Launches the simulation									
#============================================================================================


import sys, os
from com3dx import *
CATIA = get3dxClient()

#--------------------------------------------------------------------------
# 1 - Retrieves the simulation and its product
#--------------------------------------------------------------------------

myEditor = CATIA.ActiveEditor
myProdService = myEditor.GetService("PLMProductService")
myEntities = myProdService.EditedContent
myEntity = myEntities.Item(1)
MySimulationRoot = myEntity
myModel = MySimulationRoot.Model

#--------------------------------------------------------------------------
# 2 - Retrieves the Publications and Composes Links
#--------------------------------------------------------------------------

myPublications  = myModel.Publications
mySupport = myPublications.GetItem("Pump_Housing")
myTemperatureSupport1 = myPublications.GetItem("Pump_Housing_Contact_Face")
myTemperatureSupport2 = myPublications.GetItem("Pump_Housing_Internal_Surfaces")
myClampSupport = myPublications.GetItem("Pump_Housing_Contact_Face")

myDisplacementRestraintSupport1 = myPublications.GetItem("Pump_Housing_Sym_Face")
myDisplacementRestraintSupport2  = myPublications.GetItem("Pump_Housing_Sym_Face_2")
myDisplacementRestraintSupport3 = myPublications.GetItem("Pump_Housing_Sym_Face_3")

myPressureSupport = myPublications.GetItem("Pump_Housing_Internal_Surfaces")

# Composing links
# ---------------
myLinkToSupport = myProdService.ComposeLink(None, None, mySupport)
myLinkToTemperatureSupport1 = myProdService.ComposeLink(None, None, myTemperatureSupport1)
myLinkToTemperatureSupport2 = myProdService.ComposeLink(None, None, myTemperatureSupport2)
myLinkToClampSupport = myProdService.ComposeLink(None, None, myClampSupport)
myLinkToDisplacementRestraintSupport1  = myProdService.ComposeLink(None, None, myDisplacementRestraintSupport1)
myLinkToDisplacementRestraintSupport2 = myProdService.ComposeLink(None, None, myDisplacementRestraintSupport2)
myLinkToDisplacementRestraintSupport3 = myProdService.ComposeLink(None, None, myDisplacementRestraintSupport3)
myLinkToPressureSupport = myProdService.ComposeLink(None, None, myPressureSupport)

#--------------------------------------------------------------------------
# 3 - Creates the FEM Rep and retrieves its Rep Manager
#--------------------------------------------------------------------------
myPrdRepFactory =myModel.GetItem("SimPrdRepFactory")
myRepRef = myPrdRepFactory.CreatePrdRep("FEM")
myFEMRoot = myRepRef.GetItem("SimFemRoot")
myFEMRoot.AddAssociatedRep (mySupport)

#--------------------------------------------------------------------------
# 4 - Creates a Mesh on a Part
#--------------------------------------------------------------------------
myMeshSet = myFEMRoot.GetSet("SimNodesElements")
myMeshParts = myMeshSet.MeshParts
myMeshPart = myMeshParts.Add("CATFmtOctree3DMesher")
myMeshPart.AddSupport (mySupport)

myMeshPart.SetAttributeValue ("Mesh", "ElementOrder", 1)
myMeshPart.SetAttributeValue ("Mesh", "SizeValue", 4.0)
	
myMeshSet.Update()

#--------------------------------------------------------------------------
# 5 - Creates a Property
#--------------------------------------------------------------------------
myPropertiesSet = myFEMRoot.GetSet("SimProperties")
mySolidSection = myPropertiesSet.Add("SimSolidSection")

myLinkAccess = mySolidSection.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToSupport)
	
mySolidSection.MaterialBehaviorByName = "Behavior.1"

#--------------------------------------------------------------------------
# 6 - Retrieves the Scenario Manager
#--------------------------------------------------------------------------
myScenarioManager = MySimulationRoot.GetItem("SimScenarioManager")

#--------------------------------------------------------------------------
# 7 - Associates the thermal and Structural Analysis Case with the FEM rep
#--------------------------------------------------------------------------
myAnalysisCases = myScenarioManager.AnalysisCases

myThermalAnalysisCase = myAnalysisCases.Item(1)
myThermalAnalysisCase.FEMRep = myFEMRoot

myStructuralAnalysisCase = myAnalysisCases.Item(2)
myStructuralAnalysisCase.FEMRep = myFEMRoot

#--------------------------------------------------------------------------
# 8 - Creates and initializes the steady state heat transfer step
#--------------------------------------------------------------------------
myThermalSteps = myThermalAnalysisCase.Steps
mySteadyStateHeatTransferStep = myThermalSteps.Add("SimSteadyStateHeatTransferStep")
myThermalAnalysisCase.CreateDefaultElementTypeAssignment()
mySteadyStateHeatTransferStep.CreateAnalysisDefaultOutputRequests()

mySteadyStateHeatTransferStep.TotalTime = 1.0
mySteadyStateHeatTransferStep.TimeIncrementationScheme = win32com.client.constants.SimTimeIncrementation_Automatic
mySteadyStateHeatTransferStep.InitialTimeStep = 0.12
mySteadyStateHeatTransferStep.MinimumTimeStep = 0.013
mySteadyStateHeatTransferStep.MaximumTimeStep = 0.98
mySteadyStateHeatTransferStep.MaximumNumberOfIncrements = 60

#--------------------------------------------------------------------------
# 9 - Retrieves the thermal features and features states sets
#--------------------------------------------------------------------------
myThermalFeatures = myThermalAnalysisCase.Features
myThermalFeatureStates = mySteadyStateHeatTransferStep.FeatureStates

#--------------------------------------------------------------------------
# 10 - Creates thermal features
#--------------------------------------------------------------------------
myTemperature1 = myThermalFeatures.Add("SimTemperature")
myLinkAccess = myTemperature1.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToTemperatureSupport1)
myTemperature1.UniformMagnitude = 320.0
myThermalFeatureStates.CreateFeatureState (myTemperature1)

#--------------------------------------------------------------------------
# 11 - Creates and initializes the static step
#--------------------------------------------------------------------------
myStructuralSteps  = myStructuralAnalysisCase.Steps
myStaticStep  = myStructuralSteps.Add("SimStaticStep")

myStructuralAnalysisCase.CreateDefaultElementTypeAssignment()
myStaticStep.CreateAnalysisDefaultOutputRequests()

myStaticStep.MaximumIncrements = 1000

#--------------------------------------------------------------------------
# 12 - Retrieves the structural features and features states sets
#--------------------------------------------------------------------------
myStructuralFeatures  = myStructuralAnalysisCase.Features
myStructuralFeatureStates = myStaticStep.FeatureStates

#--------------------------------------------------------------------------
# 13 - Creates features
#--------------------------------------------------------------------------
# Creates a Clamp feature
# -------------------------
myClamp = myStructuralFeatures.Add("SimClamp")
myLinkAccess = myClamp.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToClampSupport)
myStructuralFeatureStates.CreateFeatureState (myClamp)

# Creating a displacement restraint feature
# ------------------------------------------
myDisplacementRestraint = myStructuralFeatures.Add("SimDisplacementRestraint")

myLinkAccess = myDisplacementRestraint.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToDisplacementRestraintSupport1)
myLinkAccess.AddLink ("MainSupport", myLinkToDisplacementRestraintSupport2)
myLinkAccess.AddLink ("MainSupport", myLinkToDisplacementRestraintSupport3)

myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimTranslation1, False)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimTranslation2, True)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimTranslation3, False)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimRotation1, True)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimRotation2, False)
myDisplacementRestraint.SetDofRestraint (win32com.client.constants.SimRotation3, True)

myStructuralFeatureStates.CreateFeatureState (myDisplacementRestraint)

# Creating a pressure feature
# ----------------------------
myPressure = myStructuralFeatures.Add("SimPressure")
myLinkAccess = myPressure.GetItem("SimLinkAccess")
myLinkAccess.AddLink ("MainSupport", myLinkToPressureSupport)

myPressure.UniformMagnitude = 500.0
myStructuralFeatureStates.CreateFeatureState (myPressure)

#--------------------------------------------------------------------------
# 14 - Launches the simulation
#--------------------------------------------------------------------------
myExecutionService = myEditor.GetService("SimExecutionService")
myExecutionService.BasicExecuteAll (MySimulationRoot)

# Switching to the results visualization workbench
# ------------------------------------------------
CATIA.StartWorkbench ("SMAHvcResultsVisualizationWorkbench")
