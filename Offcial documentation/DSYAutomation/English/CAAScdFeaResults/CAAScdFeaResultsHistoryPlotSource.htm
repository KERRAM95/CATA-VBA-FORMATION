'-------------------------------------------------------------------------- 
'---------------------------- Sample MACRO written in Basic Script ---------------------------------
'This sample shows how to create the history plot. 
 
'Steps:
'    1- Opens the simulation document and retrieves the results manager
'    2- Retrieves the results analysis case  
'    3- Creates the history plot1    
'    4- Get the step Id ,name and step description     
'    5- Sets the attributes to the history plot1
'    6- Sets the support for the history plot
'    7- Set all time based steps
'    8- Creates the history plot2    
'    9- Sets the attributes to the history plot2
'    10- Sets the step, frame and load case information
'    11- Set name to history plot
'    12- Updates the history plot
'    13- Return the minimum and maximum values
'    14- Retrieves and exports the curve


Sub CATMain()

'--------------------------------------------------------------------------
'    1- Opens the simulation document and retrieves the results manager
'--------------------------------------------------------------------------

Dim oEditor As Editor
Set oEditor = CATIA.ActiveEditor

Dim oProdService As PLMProductService
Set oProdService = oEditor.GetService("PLMProductService")

Dim oEntities As PLMEntities

If Not oProdService.EditedContent Is Nothing Then

Set oEntities = oProdService.EditedContent

Dim oEntity As PLMEntity
Dim oSimObjEntity As PLMEntity

For Each oEntity In oEntities
    If oEntity.GetCustomType = "SIMObjSimulationObjectGenericDS" Or oEntity.GetCustomType = "DesignSight" Then 
        Set oSimObjEntity = oEntity
    End If
Next

Dim oSimulationRoot As SimulationReference
Set oSimulationRoot = oSimObjEntity
    
Dim oResults As SimulationResults
Set oResults = oSimulationRoot.Results

Dim oSimulationRepRes As SimulationRepresentation
Set oSimulationRepRes = oResults.Item(1)  

Dim oRootRepRes As SimResultRepManager
Set oRootRepRes = oSimulationRepRes.Root    

Dim oResultsManager As SimResultsManager
Set oResultsManager = oRootRepRes.GetItem("SimResultsManager")   

End If

'--------------------------------------------------------------------------
'    2- Retrieves the results analysis case
'--------------------------------------------------------------------------

Dim oResultsAnalysisCases As SimResultsAnalysisCases
Set oResultsAnalysisCases = oResultsManager.ResultsAnalysisCases

Dim oResultsAnalysisCase As SimResultsAnalysisCase
Set oResultsAnalysisCase = oResultsAnalysisCases.Item(1)

'--------------------------------------------------------------------------
'    3- Creates the history plot
'--------------------------------------------------------------------------

Dim oHistoryPlot1 'As SimHistoryPlot
Set oHistoryPlot1 = oResultsAnalysisCase.CreateHistoryPlot

'--------------------------------------------------------------------------
'    4- Get the step Id ,name and step description
'--------------------------------------------------------------------------

Dim oSteps As SimResultsSteps
Set oSteps = oResultsAnalysisCase.ResultsSteps

Dim oResultsStep  As SimResultsStep
Set oResultsStep = oSteps.Item(2)

Dim stepId  As String
Dim stepName  As String
Dim stepDescription   As String
oResultsStep.GetIdentifier stepId, stepName, stepDescription

'--------------------------------------------------------------------------
'    5- Sets the attributes to the history plot1
'--------------------------------------------------------------------------

oHistoryPlot1.Variable = "U"
oHistoryPlot1.ComponentQuantity = SimVector_Component_1

'--------------------------------------------------------------------------
'    6- Sets the support for the history plot
'--------------------------------------------------------------------------

Dim MySupports(0)
MySupports(0) = "INP2SIM_NSET_Nodal Probe2"

Dim MySupports1(0)
MySupports1(0) = 3

Dim MySupports2(2)
MySupports2(0) = "Node 4"
MySupports2(1) = "Node 36"
MySupports2(2) = "Node 87"

oHistoryPlot1.SetSupport  MySupports, MySupports1,  MySupports2

'--------------------------------------------------------------------------
'    7- Set all time based steps
'--------------------------------------------------------------------------

oHistoryPlot1.SetAllTimeBasedSteps 

'--------------------------------------------------------------------------
'    8-  Creates the history plot2 
'--------------------------------------------------------------------------

Dim oHistoryPlot2 As SimHistoryPlot
Set oHistoryPlot2 = oResultsAnalysisCase.CreateHistoryPlot 
    
 '--------------------------------------------------------------------------
'    9- Sets the attributes to the history plot2
'--------------------------------------------------------------------------

oHistoryPlot2.Variable = "ALLIE"
oHistoryPlot2.ComponentQuantity =  SimScalar  

'--------------------------------------------------------------------------
'    10- Sets the step, frame and load case information
'--------------------------------------------------------------------------

oHistoryPlot2.SetStepAndFrame stepId, 0, 0

'--------------------------------------------------------------------------
'    11- Set name to history plot
'--------------------------------------------------------------------------

oHistoryPlot1.Name = "Test History Plot 1"
oHistoryPlot2.Name = "Test History Plot 2"

'--------------------------------------------------------------------------
'    12- Updates the history plot
'--------------------------------------------------------------------------

oHistoryPlot1.Update
oHistoryPlot2.Update

'--------------------------------------------------------------------------
'    13- Return the minimum and maximum values
'--------------------------------------------------------------------------

Dim dMin As Double
Dim dMax As Double

oHistoryPlot1.GetMinMaxValues dMin, dMin

'--------------------------------------------------------------------------
'    12- Retrieves and exports the curve
'--------------------------------------------------------------------------
Dim oHistCurveList()
oHistoryPlot1.GetHistoryCurveList oHistCurveList 
Dim oHistCurve As SimHistoryCurve
Set oHistCurve = oHistCurveList(0)
Dim iFileName As String
iFileName = "History_Curve1"
Dim iFileLocation As String
iFileLocation="C:\temp"
oHistCurve.Export iFileName, iFileLocation, SimXlsx

Set oHistCurve1 = oHistCurveList(1)
Dim iFileName1 As String
iFileName1 = "History_Curve_PLM"
oHistCurve1.ExportToPLMDoc iFileName1, SimCsv

End Sub
