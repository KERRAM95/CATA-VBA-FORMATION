#---------------------------- Sample script written in Python ---------------------------------
#This sample shows how to create the animation movie. There should be a field and history plot already present.
 
#Steps:
#    1- Opens the simulation document and retrieves the results manager
#    2- Retrieves the results analysis case  
#    3- Retrieves the animation option object and set the properties    
#    4- Retrieves the animation video object and creates the video     


#--------------------------------------------------------------------------
#    1- Opens the simulation document and retrieves the results manager
#--------------------------------------------------------------------------

import sys, os
from com3dx import *
CATIA = get3dxClient()
ENUM = win32com.client.constants

oEditor = CATIA.ActiveEditor

oProdService = oEditor.GetService("PLMProductService")

oEntities = oProdService.EditedContent

oEntity = oEntities.Item(1)

oSimulationRoot = oEntity 
    
oResults = oSimulationRoot.Results

oSimulationRepRes = oResults.Item(1)  

oRootRepRes = oSimulationRepRes.Root    

oResultsManager = oRootRepRes.GetItem("SimResultsManager") 

#--------------------------------------------------------------------------
#    2- Retrieves the results analysis case
#--------------------------------------------------------------------------

oResultsAnalysisCases = oResultsManager.ResultsAnalysisCases

oResultsAnalysisCase = oResultsAnalysisCases.Item(1)

#--------------------------------------------------------------------------
#    3- Retrieves the animation option object and set the properties
#--------------------------------------------------------------------------

oResAnimOptions = oResultsAnalysisCase.GetItem("SimResultsAnimationOptions")

oAnimType = ENUM.SimTimeHistory
oResAnimOptions.SetType (oAnimType)

FrameSelector = oResultsAnalysisCase.CreateFrameSelector()
FrameSelector.SymbolicFrameRange = ENUM.SimAllFrames
oResAnimOptions.SetFrameSelector (FrameSelector)

oSamplingType = ENUM.SimAllSelectedFrames
oSamplingFilterType = ENUM.SimTotalFrames
oResAnimOptions.SetSamplingRate (oSamplingType, oSamplingFilterType, 10)

eTargetSpeedType = ENUM.SimTargetSpeedMax

oResAnimOptions.SetTargetSpeed (eTargetSpeedType, 10)
oResAnimOptions.Update()
    
#--------------------------------------------------------------------------
#    4- Retrieves the animation video object and creates the video
#--------------------------------------------------------------------------

oResAnimVideo = oRootRepRes.GetItem("SimResultsAnimationVideo")

lsCodecs = oResAnimVideo.GetListOfCodecOptions (())

csCodec = "Microsoft Video 1"

oPlaybackType = ENUM.SimPlaybackTotalTime

oResAnimVideo.SetPlayback (oPlaybackType, 10)
oResAnimVideo.SetCodec (csCodec)
oResAnimVideo.SetCompressionOptions (88, True, 1, 100)
oResAnimVideo.SetXYPlotOptions (True, 40, 40, 0, 100)

ofilePath = "D:\\temp"
ofileName = "AnimationMovie.avi"
oResAnimVideo.CreateMovie (ofilePath, ofileName)
