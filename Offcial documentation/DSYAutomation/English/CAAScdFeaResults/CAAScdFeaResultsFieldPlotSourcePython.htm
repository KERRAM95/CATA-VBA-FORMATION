#---------------------------- Sample MACRO written in Basic Script ---------------------------------
#This sample creates a field plot.

#Steps:
#    1- Opens the simulation document and retrieves the results manager
#    2- Retrieves the results analysis case
#    3- Retrieves the plot from the results set
#    4- Creates the field plot through template
#    5- Sets the display group to the plot
#    6- Sets the deformation scale factor
#    7- Retrieves the legend from the plot
#    8- Retrieves the step and frame from the plot 
#    9- Deletes the plot

#--------------------------------------------------------------------------
#    1- Opens the simulation document and retrieves the results manager
#--------------------------------------------------------------------------

import sys, os
from com3dx import *
CATIA = get3dxClient()
ENUM = win32com.client.constants

oEditor = CATIA.ActiveEditor

oProdService = oEditor.GetService("PLMProductService")

oEntities = oProdService.EditedContent

oEntity = oEntities.Item(1)

oSimulationRoot = oEntity 
    
oResults = oSimulationRoot.Results

oSimulationRepRes = oResults.Item(1)  

oRootRepRes = oSimulationRepRes.Root    

oResultsManager = oRootRepRes.GetItem("SimResultsManager") 

#--------------------------------------------------------------------------
#    2- Retrieves the results analysis case
#--------------------------------------------------------------------------

oResultsAnalysisCases = oResultsManager.ResultsAnalysisCases
oResultsAnalysisCase = oResultsAnalysisCases.Item(1)

#--------------------------------------------------------------------------
#    3- Retrieves the plot from the results set
#--------------------------------------------------------------------------

oFieldPlots = oResultsAnalysisCase.GetSet("FieldPlot")

oFieldPlots = oFieldPlots[0]
oFieldPlot = oFieldPlots.Item(1)

#--------------------------------------------------------------------------
#    4- Retrieves plot ID#s and creates the field plots through template
#--------------------------------------------------------------------------
    
sIdList = oResultsAnalysisCase.GetListOfFieldPlotIDs ((), True, True, ())

sId1 = sIdList[2]
oPlot = oResultsAnalysisCase.CreateFieldPlotByID(sId1, True)
oPlot = oPlot[0]
sId2 = "Contour_Von_Mises"
oPlot1 = oResultsAnalysisCase.CreateFieldPlotByID(sId2, False)
oPlot1 = oPlot1[0]

#--------------------------------------------------------------------------
#    5- Sets support to the sensor
#--------------------------------------------------------------------------

oDisplayGroupSet = oResultsAnalysisCase.GetSet("DisplayGroupSet")
oDisplayGroupSet = oDisplayGroupSet[0]
oDisplayGroup = oDisplayGroupSet.Item(1)
             
MyDispatch = oDisplayGroup
MySupports = [MyDispatch]
oPlot.DisplayGroup = (MyDispatch)
oPlot.Refresh()

#--------------------------------------------------------------------------
#    6- Sets the deformation scale factor
#--------------------------------------------------------------------------

oPlot.SetDeformationScaleFactors (2, 2, 2)

#--------------------------------------------------------------------------
#    7- Retrieves the legend form the plot
#--------------------------------------------------------------------------

oPlotLegend = oPlot.Legend

#--------------------------------------------------------------------------
#    8- Retrieves the step and frame from the plot
#--------------------------------------------------------------------------

nbFrame = oPlot.GetCurrentFrame (True)

sStep, nbframe, nbLoadCase = oPlot.GetStepAndFrame ("", 0, 0, False)

#--------------------------------------------------------------------------
#    9- Deletes the plot
#--------------------------------------------------------------------------

oResultsAnalysisCase.DeleteFeature (oPlot1)
